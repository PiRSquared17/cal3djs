 18 /**
 19 	@class CalAnimation
 20 */
 21 Cal3D.CalAnimation = function(coreAnimation) {
 22 	this.m_coreAnimation = coreAnimation;
 23 	this.m_lastCallbackTimes = [];
 24 	this.m_type = Cal3D.CalAnimation.Type.TYPE_NONE;
 25 	this.m_state = Cal3D.CalAnimation.State.STATE_NONE;
 26 	this.m_time = 0;
 27 	this.m_timeFactor = 1;
 28 	this.m_weight = 0;
 29 
 30 	// build up the last called list
 31 	var callbackList = coreAnimation.getCallbackList();
 32 	for(var i=0; i<callbackList.length; i++) {
 33 		this.m_lastCallbackTimes.push(0);
 34 	}
 35 };

143 /**
144 	@private
145 */
146 Cal3D.CalAnimation.prototype.checkCallbacks = function(animationTime, model) {
147 	var callbackList = this.m_coreAnimation.getCallbackList();
148 	for (var i=0; i<callbackList.length; i++) {
149 		// support for dynamic adding of callbacks
150 		if (this.m_lastCallbackTimes.length <= i)
151 			this.m_lastCallbackTimes.push(animationTime);
152 
153 		callbackList[i].callback.AnimationUpdate(animationTime, model, model.getUserData());
154 		if (animationTime > 0 && animationTime < this.m_lastCallbackTimes[i])		// looped
155 			this.m_lastCallbackTimes[i] -= this.m_coreAnimation.getDuration();
156 		else if (animationTime < 0 && animationTime > this.m_lastCallbackTimes[i])	// reverse-looped  
157 			this.m_lastCallbackTimes[i] += this.m_coreAnimation.getDuration();
158 
159 		if ( (animationTime >= 0 && animationTime >= this.m_lastCallbackTimes[i] + callbackList[i].min_interval) ||
160 			 (animationTime <  0 && animationTime <= this.m_lastCallbackTimes[i] - callbackList[i].min_interval) ) {
161 			callbackList[i].callback.AnimationUpdate(animationTime, model, model.getUserData());
162 			this.m_lastCallbackTimes[i] = animationTime;
163 		}
164 	}
165 };
166 
167 /**
168 	@private
169 */
170 Cal3D.CalAnimation.prototype.completeCallbacks = function(model) {
171 	var callbackList = this.m_coreAnimation.getCallbackList();
172 	for(var i=0; i<callbackList.length; i++) {
173 		callbackList[i].callback.AnimationComplete(model, model.getUserData());
174 	}
175 };

216 /**
217 	@class CalAnimationAction
218 */
219 Cal3D.CalAnimationAction = function(coreAnimation) {
220 	// call constructor of super class
221 	Cal3D.CalAnimation.call(this, coreAnimation);
222 
223 	this.m_delayIn = 0;
224 	this.m_delayOut = 0;
225 	this.m_delayTarget = 0;
226 	this.m_weightTarget = 0;
227 	this.m_autoLock = false;
228 
229 	this.setType(Cal3D.CalAnimation.Type.TYPE_ACTION);
230 };
231 
232 // inherited from CalAnimation
233 Cal3D.CalAnimationAction.prototype = new Cal3D.CalAnimation(new Cal3D.CalCoreAnimation /* dummy object */);
234 
235 /**
236 	Execute the animation action instance.<br />
237 	This function executes the animation action instance.
238 	@param {number} delayIn The time in seconds until the animation action instance reaches the full weight from the beginning of its execution.
239 	@param {number} delayOut The time in seconds in which the animation action instance reaches zero weight at the end of its execution.
240 	@param {number} weightTarget Target weight.
241 	@param {boolean} autoLock If set to true, the animation will not be reset and removed on the last keyframe.
242 */
243 Cal3D.CalAnimationAction.prototype.execute = function(delayIn, delayOut, weightTarget, autoLock) {
244 	if(weightTarget == undefined)
245 		weightTarget = 1;
246 	if(autoLock == undefined)
247 		autoLock = false;
248 
249 	this.setState(Cal3D.CalAnimation.State.STATE_IN);
250 	this.setWeight(0);
251 	this.m_delayIn = delayIn;
252 	this.m_delayOut = delayOut;
253 	this.setTime(0);
254 	this.m_weightTarget = weightTarget;
255 	this.m_autoLock = autoLock;
256 
257 	return true;
258 };
259 
260 /**
261 	Update the animation action instance.<br />
262 	This function updates the animation action instance for a given amount of time.
263 	@param {number} deltaTime The elapsed time in seconds since the last update.
264 	@returns {boolean} Whether or not the end the the action has been reached. 
265 		<li>true, if the animation action instance is still active;</li>
266 		<li>false, if the execution of the animation action instance has ended.</li>
267 */
268 Cal3D.CalAnimationAction.prototype.update = function(deltaTime) {
269 	// update animation action time
270 	if(this.getState() != Cal3D.CalAnimation.State.STATE_STOPPED) {
271 		this.setTime(this.getTime() + deltaTime * this.getTimeFactor());
272 	}
273 
274 	// handle IN phase
275 	if(this.getState() == Cal3D.CalAnimation.State.STATE_IN) {
276 		// check if we are still in the IN phase
277 		if(this.getTime() < this.m_delayIn) {
278 			this.setWeight(this.getTime() / this.m_delayIn * this.m_weightTarget);
279 		}
280 		else {
281 			this.setState(Cal3D.CalAnimation.State.STATE_STEADY);
282 			this.setWeight(this.m_weightTarget);
283 		}
284 	}
285 
286 	// handle STEADY
287 	if(this.getState() == Cal3D.CalAnimation.State.STATE_STEADY) {
288 		// check if we reached OUT phase
289 		if(!this.m_autoLock && this.getTime() >= this.getCoreAnimation().getDuration() - this.m_delayOut) {
290 			this.setState(Cal3D.CalAnimation.State.STATE_OUT);
291 		}
292 		// if the anim is supposed to stay locked on last keyframe, reset the time here.
293 		else if(this.m_autoLock && this.getTime() > this.getCoreAnimation().getDuration()) {
294 			this.setState(Cal3D.CalAnimation.State.STATE_STOPPED);
295 			this.setTime(this.getCoreAnimation().getDuration());
296 		}      
297 	}
298 
299 	// handle OUT phase
300 	if(this.getState() == Cal3D.CalAnimation.State.STATE_OUT) {
301 		// check if we are still in the OUT phase
302 		if(this.getTime() < this.getCoreAnimation().getDuration()) {
303 			this.setWeight((this.getCoreAnimation().getDuration() - this.getTime()) / this.m_delayOut * this.m_weightTarget);
304 		}
305 		else {
306 			// we reached the end of the action animation
307 			this.setWeight(0);
308 			return false;
309 		}
310 	}
311 
312 	return true;
313 };

317 /**
318 	@class CalAnimationCycle
319 */
320 Cal3D.CalAnimationCycle = function(coreAnimation) {
321 	// call constructor of super class
322 	Cal3D.CalAnimation.call(this, coreAnimation);
323 
324 	this.m_targetDelay = 0;
325 	this.m_targetWeight = 0;
326 
327 	this.setType(Cal3D.CalAnimation.Type.TYPE_CYCLE);
328 	this.setState(Cal3D.CalAnimation.State.STATE_SYNC);
329 
330 	this.setWeight(0);
331 };

333 // inherited from CalAnimation
334 Cal3D.CalAnimationCycle.prototype = new Cal3D.CalAnimation(new Cal3D.CalCoreAnimation /* dummy object */);

336 /**
337 	Interpolate the weight of the animation cycle instance.<br />
338 	This function interpolates the weight of the animation cycle instance to a new value in a given amount of time.
339 	@param {number} weight The weight to interpolate the animation cycle instance to. 
340 	@param {number} delay The time in seconds until the new weight should be reached. 
341 	@returns {boolean} true if succeeded; false if error happened.
342 */
343 Cal3D.CalAnimationCycle.prototype.blend = function(weight, delay) {
344   this.m_targetWeight = weight;
345   this.m_targetDelay = delay;
346 
347   return true;
348 };
349 
350 /**
351 	Put the animation cycle instance into async state.<br />
352 	This function puts the animation cycle instance into async state, which means that it will end after the current running cycle.
353 	@param {number} time The time in seconds at which the animation cycle instance was unlinked from the global mixer animation cycle.
354 	@param {number} duration The current duration of the global mixer animation cycle in seconds at the time of the unlinking. 
355 */
356 Cal3D.CalAnimationCycle.prototype.setAsync = function(time, duration) {
357 	// check if thie animation cycle is already async
358 	if(this.getState() != Cal3D.CalAnimation.State.STATE_ASYNC) {
359 		if(duration == 0) {
360 			this.setTimeFactor(1);
361 			this.setTime(0);
362 		}
363 		else {
364 			this.setTimeFactor(this.getCoreAnimation().getDuration() / duration);
365 			this.setTime(time * this.getTimeFactor());
366 		}
367 
368 		this.setState(Cal3D.CalAnimation.State.STATE_ASYNC);
369 	}
370 };
371 
372 /**
373 	Update the animation cycle instance.<br />
374 	This function updates the animation cycle instance for a given amount of time.
375 	@param {number} deltaTime The elapsed time in seconds since the last update.
376 	@returns {boolean} Whether or not the animation cycle has ended.
377 		<li>true if the animation cycle instance is still active.</li>
378 		<li>false if the execution of the animation cycle instance has ended.</li>
379 */
380 Cal3D.CalAnimationCycle.prototype.update = function(deltaTime) {
381 	if(this.m_targetDelay <= Math.abs(deltaTime)) {
382 		// we reached target delay, set to full weight
383 		this.setWeight(this.m_targetWeight);
384 		this.m_targetDelay = 0;
385 
386 		// check if we reached the cycles end
387 		if(this.getWeight() == 0) {
388 			return false;
389 		}
390 	}
391 	else {
392 		// not reached target delay yet, interpolate between current and target weight
393 		var factor = deltaTime / this.m_targetDelay;
394 		this.setWeight((1 - factor) * this.getWeight() + factor * this.m_targetWeight);
395 		this.m_targetDelay -= deltaTime;
396 	}
397 
398 	// update animation cycle time if it is in async state
399 	if(this.getState() == Cal3D.CalAnimation.State.STATE_ASYNC) {
400 		this.setTime(this.getTime() + deltaTime * this.getTimeFactor());
401 		if(this.getTime() >= this.getCoreAnimation().getDuration()) {
402 			this.setTime(this.getTime() % this.getCoreAnimation().getDuration());
403 		}
404 		if (this.getTime() < 0) {
405 			this.setTime(this.getTime() + this.getCoreAnimation().getDuration());
406 		}
407 	}
408 
409 	return true;
410 };
