 18 /**
 19 	@class CalCoreTrack
 20 */
 21 Cal3D.CalCoreTrack = function() {
 22 	this.m_coreBoneId = -1;
 23 	this.m_keyframes = [];
 24 };
 25 
 26 /**
 27 	Create the core track.
 28 	@returns {boolean} true if succeeded; false if not.
 29 */
 30 Cal3D.CalCoreTrack.prototype.create = function() {
 31 	return true;
 32 };
 33 
 34 /**
 35 	Destroy the core track.
 36 */
 37 Cal3D.CalCoreTrack.prototype.destroy = function() {
 38 	// destroy all core key-frames
 39 	for(var i=0; i<this.m_keyframes.length; i++) {
 40 		this.m_keyframes[i].destroy();
 41 	}
 42 
 43 	this.m_keyframes = [];
 44 	this.m_coreBoneId = -1;
 45 };
 46 
 47 /**
 48 	Get a state.<br />
 49 	This function returns the state (translation and rotation of the core bone) for the specified time and duration.
 50 	@param {number} time The time in seconds at which the state should be returned.
 51 	@param {Cal3D.CalVector} translation A vector as an output parameter that will be fiiled with specified state.
 52 	@param {Cal3D.CalQuaternion} rotation A quaternion as an output parameter that will be fiiled with specified state.
 53 	@returns {boolean} true if succeeded; false if any error happened.
 54 */
 55 Cal3D.CalCoreTrack.prototype.getState = function(time, translation, rotation) {
 56 	var coreKeyframeIdBefore, coreKeyframeIdAfter;
 57 
 58 	// get the key-frame after the requested time
 59 	coreKeyframeIdAfter = this.getUpperBound(time);
 60 
 61 	// check if the time is after the last key-frame
 62 	if(coreKeyframeIdAfter == this.m_keyframes.length) {
 63 		// return the last key-frame state
 64 		coreKeyframeIdAfter--;
 65 		rotation.assign(this.m_keyframes[coreKeyframeIdAfter].getRotation());
 66 		translation.assign(this.m_keyframes[coreKeyframeIdAfter].getTranslation());
 67 		return true;
 68 	}
 69 
 70 	// check if the time is before the first key-frame
 71 	if(coreKeyframeIdAfter == 0) {
 72 		// return the first keyframe state
 73 		rotation.assign(this.m_keyframes[coreKeyframeIdAfter].getRotation());
 74 		translation.assign(this.m_keyframes[coreKeyframeIdAfter].getTranslation());
 75 		return true;
 76 	}
 77 
 78 	// get the key-frame before the requested one
 79 	coreKeyframeIdBefore = coreKeyframeIdAfter - 1;
 80 
 81 	// get the two key-frames
 82 	var coreKeyframeBefore, coreKeyframeAfter;
 83 	coreKeyframeBefore = this.m_keyframes[coreKeyframeIdBefore];
 84 	coreKeyframeAfter = this.m_keyframes[coreKeyframeIdAfter];
 85 
 86 	// calculate the blending factor between the two key-frame states
 87 	var blendFactor = (time - coreKeyframeBefore.getTime()) / (coreKeyframeAfter.getTime() - coreKeyframeBefore.getTime());
 88 
 89 	/* 
 90 		blend between the two key-frames
 91 	*/
 92 	translation.assign(coreKeyframeBefore.getTranslation());
 93 	translation.blend(blendFactor, coreKeyframeAfter.getTranslation());
 94 
 95 	rotation.assign(coreKeyframeBefore.getRotation());
 96 	rotation.blend(blendFactor, coreKeyframeAfter.getRotation());
 97 
 98 	return true;
 99 };

101 /**
102 	Get the ID of the core bone.<br />
103 	This function returns the ID of the core bone to which the core track instance is attached to.
104 	@returns {number} The ID of the core bone.
105 */
106 Cal3D.CalCoreTrack.prototype.getCoreBoneId = function() {
107 	return this.m_coreBoneId;
108 };
109 
110 /**
111 	Set the ID of the core bone.<br />
112 	This function sets the ID of the core bone to which the core track instance is attached to.
113 	@param {number} coreBoneId  The ID of the core bone to which the core track should be attached to.
114 	@returns {boolean} true if succeeded; false if the core bone does not exist.
115 */
116 Cal3D.CalCoreTrack.prototype.setCoreBoneId = function(coreBoneId) {
117 	if(coreBoneId < 0) {
118 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_HANDLE, 'coretrack.js');
119 		return false;
120 	}
121 
122 	this.m_coreBoneId = coreBoneId;
123 	return true;
124 };
125 
126 /**
127 	Get the number of core keyframes.<br />
128 	@returns {number} The number of core keyframes.
129 */
130 Cal3D.CalCoreTrack.prototype.getCoreKeyframeCount = function() {
131 	return this.m_keyframes.length;
132 };
133 
134 /**
135 	Get a core keyframe.<br />
136 	@param {number} index The index of the core keyframe.
137 	@returns {Cal3D.CalCoreKeyframe} The core keyframe.
138 */
139 Cal3D.CalCoreTrack.prototype.getCoreKeyframe = function(index) {
140 	return this.m_keyframes[index];
141 };
142 
143 /**
144 	Add a core keyframe.<br />
145 	This function adds a core keyframe to the core track instance.
146 	@param {Cal3D.CalCoreKeyframe} coreKeyframe The core keyframe to be added.
147 	@returns {boolean} ID of the added core keyframe.
148 */
149 Cal3D.CalCoreTrack.prototype.addCoreKeyframe = function(coreKeyframe) {
150 	this.m_keyframes.push(coreKeyframe);
151 
152 	var index = this.m_keyframes.length - 1;
153 	while(index > 0 && this.m_keyframes[index].getTime() < this.m_keyframes[index-1].getTime()) {
154 		var temp = this.m_keyframes[index];
155 		this.m_keyframes[index] = this.m_keyframes[index - 1];
156 		this.m_keyframes[index - 1] = temp;
157 		index--;
158 	}
159 
160 	return true;
161 };
162 
163 /**
164 	Remove a core keyframe from the core track.<br />
165 	This function removes a core keyframe from the core track instance.
166 	@param {number} index The index of the core keyframe to be removed.
167 */
168 Cal3D.CalCoreTrack.prototype.removeCoreKeyFrame = function(index) {
169 	this.m_keyframes.splice(index, 1);
170 };
171 
172 /**
173 	Scale the core track.<br />
174 	This function rescale all the data that are in the core track instance.
175 	@param {number} factor The scale factor.
176 */
177 Cal3D.CalCoreTrack.prototype.scale = function(factor) {
178 	for(var keyframeId=0; keyframeId<this.m_keyframes.length; keyframeId++) {
179 		this.m_keyframes[keyframeId].getTranslation().multScalarLocal(factor);
180 	}
181 };
182 
183 /**
184 	@private
185 */
186 Cal3D.CalCoreTrack.prototype.getUpperBound = function(time) {
187 	var lower = 0;
188 	var upper = this.m_keyframes.length - 1;
189 
190 	while(lower < upper - 1) {
191 		var middle = (lower + upper) >> 1;
192 
193 		if(time >= this.m_keyframes[middle].getTime()) {
194 			lower = middle;
195 		}
196 		else {
197 			upper = middle;
198 		}
199 	}
200 
201 	return upper;
202 };
