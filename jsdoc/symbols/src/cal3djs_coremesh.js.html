 18 /**
 19 	@class CalCoreMesh
 20 */
 21 Cal3D.CalCoreMesh = function() {
 22 	this.m_vectorCoreSubmesh = [];
 23 	this.m_name = '';
 24 	this.m_filename = '';
 25 };
 26 
 27 /**
 28 	Add a core submesh to the core mesh.<br />
 29 	This function adds a core submesh to the core mesh instance.
 30 	@param {Cal3D.CalCoreSubmesh} coreSubmesh The core submesh to be added.
 31 	@returns {number} ID of the added submesh. 
 32 */
 33 Cal3D.CalCoreMesh.prototype.addCoreSubmesh = function(coreSubmesh) {
 34 	var submeshId = this.m_vectorCoreSubmesh.length;
 35 	this.m_vectorCoreSubmesh.push(coreSubmesh);
 36 	return submeshId;
 37 };
 38 
 39 /**
 40 	Get a core submesh.<br />
 41 	This function returns the core submesh with the given ID.
 42 	@param {number} coreSubmeshId ID of the core submesh.
 43 	@returns {Cal3D.CalCoreSubmesh} The core submesh instance; null if the core submesh does not exist.
 44 */
 45 Cal3D.CalCoreMesh.prototype.getCoreSubmesh = function(coreSubmeshId) {
 46 	if(coreSubmeshId < 0 || coreSubmeshId >= this.m_vectorCoreSubmesh.length) {
 47 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_HANDLE, 'coremesh.js');
 48 		return null;
 49 	}
 50 
 51 	return this.m_vectorCoreSubmesh[coreSubmeshId];
 52 };
 53 
 54 /**
 55 	Get the number of core submeshes.<br />
 56 	This function returns the number of core submeshes in the core mesh instance.
 57 	@returns {number} The number of core submeshes.
 58 */
 59 Cal3D.CalCoreMesh.prototype.getCoreSubmeshCount = function() {
 60 	return this.m_vectorCoreSubmesh.length;
 61 };
 62 
 63 /**
 64 	Get the core submesh array.<br />
 65 	This function returns the array that contains all core submeshes of the core mesh instance.
 66 	@returns {Array} The core submeshes array.
 67 */
 68 Cal3D.CalCoreMesh.prototype.getVectorCoreSubmesh = function() {
 69 	return this.m_vectorCoreSubmesh;
 70 };
 71 
 72 /**
 73 	Add a blend target.<br />
 74 	This function adds a core mesh as a blend target. <br />
 75 	It adds appropriate CalCoreSubMorphTargets to each of the core submeshes.
 76 	@param {Cal3D.CalCoreMesh} the core mesh that should become a blend target.
 77 	@returns {number} ID of the added blend target; -1 if any error happened.
 78 */
 79 Cal3D.CalCoreMesh.prototype.addAsMorphTarget = function(coreMesh) {
 80 	/*
 81 		Check if the numbers of vertices allow a blending
 82 	*/
 83 
 84 	var otherVectorCoreSubmesh = coreMesh.getVectorCoreSubmesh();
 85 	if(this.m_vectorCoreSubmesh.length != otherVectorCoreSubmesh.length) {
 86 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INTERNAL, 'coremesh.js');
 87 		return -1;
 88 	}
 89 	if(this.m_vectorCoreSubmesh.length == 0) {
 90 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INTERNAL, 'coremesh.js');
 91 		return -1;
 92 	}
 93 
 94 	var submeshId = 0;
 95 	var subMorphTargetId = this.m_vectorCoreSubmesh[submeshId].getCoreSubMorphTargetCount();
 96 	while(submeshId < this.m_vectorCoreSubmesh.length) {
 97 		if(this.m_vectorCoreSubmesh[submeshId].getVertexCount() != otherVectorCoreSubmesh[submeshId].getVertexCount()) {
 98 			Cal3D.CalError.setLastError(Cal3D.CalError.Code.INTERNAL, 'coremesh.js');
 99 			return -1;
100 		}
101 		submeshId++;
102 	}
103 
104 	/* 
105 		Add the blend targets to each of the core sub meshes
106 	*/
107 
108 	submeshId = 0;
109 	while(submeshId < this.m_vectorCoreSubmesh.length) {
110 		var vertexCount = otherVectorCoreSubmesh[submeshId].getVertexCount();
111 		var coreSubMorphTarget = new Cal3D.CalCoreSubMorphTarget;
112 		if(!coreSubMorphTarget.reserve(vertexCount))
113 			return -1;
114 
115 		var vectorVertex = otherVectorCoreSubmesh[submeshId].getVectorVertex();
116 		var vertexId = 0;
117 		for(var i=0; i<vertexCount; i++) {
118 			var blendVertex = new Cal3D.CalCoreSubMorphTarget.BlendVertex;
119 			blendVertex.position.assign(vectorVertex[vertexId].position);
120 			blendVertex.normal.assign(vectorVertex[vertexId].normal);
121 			if(!coreSubMorphTarget.setBlendVertex(i, blendVertex))
122 				return -1;
123 
124 			vertexId++;
125 		}
126 
127 		this.m_vectorCoreSubmesh[submeshId].addCoreSubMorphTarget(coreSubMorphTarget);
128 		submeshId++;
129 	}
130 
131 	return subMorphTargetId;
132 };
133 
134 /**
135 	Scale the Mesh.<br />
136 	This function rescale all the data that are in the core mesh instance.
137 	@param {number} factor The scale factor.
138 */
139 Cal3D.CalCoreMesh.prototype.scale = function(factor) {
140 	for(var submeshId=0; submeshId<this.m_vectorCoreSubmesh.length; submeshId++) {
141 		this.m_vectorCoreSubmesh[submeshId].scale(factor);
142 	}
143 };

145 /**
146 	Set the name of the file in which the core mesh is stored, if any.<br />
147 	@param {string} filename The file name.
148 */
149 Cal3D.CalCoreMesh.prototype.setFilename = function(filename) {
150 	this.m_filename = filename;
151 };
152 
153 /**
154 	Get the name of the file in which the core mesh is stored, if any.<br />
155 	@returns {string} The file name. An empty string will be returned if the mesh is not stored in a file.
156 */
157 Cal3D.CalCoreMesh.prototype.getFilename = function() {
158 	return this.m_filename;
159 };
160 
161 /**
162 	Set the symbolic name of the core mesh.<br />
163 	@param {string} name The symbolic name to be set.
164 */
165 Cal3D.CalCoreMesh.prototype.setName = function(name) {
166 	this.m_name = name;
167 };
168 
169 /**
170 	Get the symbolic name of the core mesh.<br />
171 	@returns {string} The symbolic name of the core mesh. an empty string will be returned if the symbolic name is not set yet.
172 */
173 Cal3D.CalCoreMesh.prototype.getName = function() {
174 	return this.m_name;
175 };
