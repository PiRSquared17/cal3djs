 18 /**
 19 	@class CalMesh
 20 */
 21 Cal3D.CalMesh = function(coreMesh) {
 22 	this.m_model = null;
 23 	this.m_coreMesh = coreMesh;
 24 	this.m_vectorSubmesh = []
 25 
 26 	// clone the mesh structure of the core mesh
 27 	var vectorCoreSubmesh = coreMesh.getVectorCoreSubmesh();
 28 
 29 	// get the number of submeshes
 30 	var submeshCount = vectorCoreSubmesh.length;
 31 
 32 	// clone every core submesh
 33 	for(var submeshId=0; submeshId<submeshCount; submeshId++) {
 34 		this.m_vectorSubmesh.push(new Cal3D.CalSubmesh(vectorCoreSubmesh[submeshId]));
 35 	}
 36 };
 37 
 38 /**
 39 	Get the core mesh.<br />
 40 	This function returns the core mesh on which this mesh instance is based on.
 41 	@returns {Cal3D.CalCoreMesh} The core mesh.
 42 */
 43 Cal3D.CalMesh.prototype.getCoreMesh = function() {
 44 	return this.m_coreMesh;
 45 };
 46 
 47 /**
 48 	Get a submesh.<br />
 49 	This function returns the submesh with the given ID.
 50 	@param {number} submeshId The ID of the submesh.
 51 	@returns {Cal3D.CalSubmesh} The submesh; null if it does not exist. 
 52 */
 53 Cal3D.CalMesh.prototype.getSubmesh = function(submeshId) {
 54 	if(submeshId < 0 || submeshId >= this.m_vectorSubmesh.length) {
 55 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_HANDLE, 'mesh.js');
 56 		return null;
 57 	}
 58 
 59 	return this.m_vectorSubmesh[submeshId];
 60 };
 61 
 62 /**
 63 	Get the number of submeshes.<br />
 64 	This function returns the number of submeshes in the mesh instance.
 65 	@returns {number} The number of submeshes.
 66 */
 67 Cal3D.CalMesh.prototype.getSubmeshCount = function() {
 68 	return this.m_vectorSubmesh.length;
 69 };
 70 
 71 /**
 72 	Get the submesh list.<br />
 73 	This function returns the list that contains all submeshes of the mesh instance.
 74 	@returns {Array} The submesh list.
 75 */
 76 Cal3D.CalMesh.prototype.getVectorSubmesh = function() {
 77 	return this.m_vectorSubmesh;
 78 };
 79 
 80 /**
 81 	Set the current LOD level.<br />
 82 	This function sets the current LOD level of the mesh instance.
 83 	@param {number} lodLevel The LOD level in the range [0.0, 1.0].
 84 */
 85 Cal3D.CalMesh.prototype.setLodLevel = function(lodLevel) {
 86 	// change lod level of each submesh
 87 	for(var submeshId=0; submeshId<this.m_vectorSubmesh.length; submeshId++) {
 88 		// set the lod level in the submesh
 89 		this.m_vectorSubmesh[submeshId].setLodLevel(lodLevel);
 90 	}
 91 };
 92 
 93 /**
 94 	Set the material set.<br />
 95 	This function sets the material set of the mesh instance.
 96 	@param {number} setId The ID of the material set.
 97 */
 98 Cal3D.CalMesh.prototype.setMaterialSet = function(setId) {
 99 	// change material of each submesh
100 	for(var submeshId=0; submeshId<this.m_vectorSubmesh.length; submeshId++) {
101 		// get the core material thread id of the submesh
102 		var coreMaterialThreadId = this.m_vectorSubmesh[submeshId].getCoreSubmesh().getCoreMaterialThreadId();
103 
104 		// get the core material id for the given set id in the material thread
105 		var coreMaterialId = this.m_model.getCoreModel().getCoreMaterialId(coreMaterialThreadId, setId);
106 
107 		// set the new core material id in the submesh
108 		this.m_vectorSubmesh[submeshId].setCoreMaterialId(coreMaterialId);
109 	}
110 };
111 
112 /**
113 	Set the model.<br />
114 	This function sets the model to which the mesh instance is attached to.
115 	@param {Cal3D.CalModel} model The model to which the mesh instance should be attached to.
116 */
117 Cal3D.CalMesh.prototype.setModel = function(model) {
118 	this.m_model = model;
119 };
120 
121 /**
122 	Disable internal data (and thus spring system).
123 */
124 Cal3D.CalMesh.prototype.disableInternalData = function() {
125 	// disable internal data of each submesh
126 	for(var submeshId=0; submeshId<this.m_vectorSubmesh.length; submeshId++) {
127 		// disable internal data of the submesh
128 		this.m_vectorSubmesh[submeshId].disableInternalData();
129 	}
130 };
