 53 /**
 54 	Scale the core model.<br />
 55 	This function rescale all data that are in the core model instance.
 56 	@param {number} factor The scale factor.
 57 */
 58 Cal3D.CalCoreModel.prototype.scale = function(factor) {
 59 	this.m_coreSkeleton.scale(factor);
 60 
 61 	for(var animationId=0; animationId<this.m_vectorCoreAnimation.length; animationId++) {
 62 		this.m_vectorCoreAnimation[animationId].scale(factor);
 63 	}
 64 
 65 	for(var meshId=0; meshId<this.m_vectorCoreMesh.length; meshId++) {
 66 		this.m_vectorCoreMesh[meshId].scale(factor);
 67 	}
 68 };

110 /**
111 	Delete the resources used by the named core animation.<br />
112 	The name must be associated with a valid core animation ID with the function getAnimationId(). The caller must ensure that the 
113 	corresponding is not referenced anywhere otherwise unpredictable results will occur.
114 	@param {number | string} ID or name of the core animation.
115 	@returns {number} ID of the unloaded core animation; -1 if the core animation does not exist.
116 */
117 Cal3D.CalCoreModel.prototype.unloadCoreAnimation = function(coreAnimationId) {
118 	if((typeof coreAnimationId) == 'string')
119 		coreAnimationId = this.getCoreAnimationId(coreAnimationId);
120 
121 	if(coreAnimationId < 0 || coreAnimationId >= this.m_vectorCoreAnimation.length) {
122 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_HANDLE, 'coremodel.js');
123 		return -1;
124 	}
125 
126 	this.m_vectorCoreAnimation[coreAnimationId] = null;
127 	return coreAnimationId;
128 };

267 /**
268 	Get a specified core material ID.<br />
269 	This function returns the core material ID for a specified core material name or thread/core material set pair.
270 	@param {string | number} materialName The name that is associated with a core material ID, or the ID of the core material thread.
271 	@param {number} coreMaterialSetId (Optional) The ID of the core material set.
272 	@returns {number} ID of the core material; -1 if it does not exist.
273 */
274 Cal3D.CalCoreModel.prototype.getCoreMaterialId = function(materialName) {
275 	if(arguments.length == 1) {
276 		var coreMaterialId = this.m_materialName[materialName];
277 		if(coreMaterialId == undefined)
278 			return -1;
279 
280 		if(!this.getCoreMaterial(coreMaterialId))
281 			return -1;
282 
283 		return coreMaterialId;
284 	}
285 	else if(arguments.length == 2) {
286 		var coreMaterialThreadId = arguments[0];
287 		var coreMaterialSetId = arguments[1];
288 
289 		// find the core material thread
290 		var coreMaterialThread = this.m_mapmapCoreMaterialThread[coreMaterialThreadId];
291 		if(coreMaterialThread == undefined) {
292 			Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_HANDLE, 'coremodel.js');
293 			return -1;
294 		}
295 
296 		// find the core material id for the given set
297 		var coreMaterialId = coreMaterialThread[coreMaterialSetId];
298 		if(coreMaterialId == undefined) {
299 			Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_HANDLE, 'coremodel.js');
300 			return -1;
301 		}
302 
303 		return coreMaterialId;
304 	}
305 
306 	return -1;
307 };
