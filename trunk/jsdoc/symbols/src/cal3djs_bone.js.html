 18 /**
 19 	@class CalBone
 20 */
 21 Cal3D.CalBone = function(coreBone) {
 22 	this.m_coreBone = coreBone;
 23 	this.m_skeleton = null;
 24 	this.m_accumulatedWeight = 0;
 25 	this.m_accumulatedWeightAbsolute = 0;
 26 	this.m_translation = new Cal3D.CalVector;
 27 	this.m_rotation = new Cal3D.CalQuaternion;
 28 	this.m_translationAbsolute = new Cal3D.CalVector;
 29 	this.m_rotationAbsolute = new Cal3D.CalQuaternion;
 30 	this.m_translationBoneSpace = new Cal3D.CalVector;
 31 	this.m_rotationBoneSpace = new Cal3D.CalQuaternion;
 32 	this.m_transformMatrix = new Cal3D.CalMatrix;
 33 	this.m_boundingBox = new Cal3D.CalBoundingBox;
 34 
 35 	this.clearState();
 36 };
 37 
 38 /**
 39 	Interpolate the current state to another state.<br />
 40 	This function interpolates the current state (relative translation and rotation) of the bone instance to another state of a given weight.
 41 	@param {number} weight The blending weight.
 42 	@param {Cal3D.CalVector} translation The relative translation to be interpolated to.
 43 	@param {Cal3D.CalQuaternion} rotation The relative rotation to be interpolated to.
 44 */
 45 Cal3D.CalBone.prototype.blendState = function(weight, translation, rotation) {
 46 	if(this.m_accumulatedWeightAbsolute == 0) {
 47 		// it is the first state, so just copy it into the bone state
 48 		this.m_translationAbsolute.assign(translation);
 49 		this.m_rotationAbsolute.assign(rotation);
 50 		this.m_accumulatedWeightAbsolute = weight;
 51 	}
 52 	else {
 53 		// it is not the first state, so blend all attributes
 54 		var factor = weight / (this.m_accumulatedWeightAbsolute + weight);
 55 
 56 		this.m_translationAbsolute.blend(factor, translation);
 57 		this.m_rotationAbsolute.blend(factor, rotation);
 58 		this.m_accumulatedWeightAbsolute += weight;
 59 	}
 60 };

 62 /**
 63 	Calculate the current state.<br />
 64 	This function calculates the current state (absolute translation and rotation, as well as the bone space transformation) of 
 65 	the bone instance and all its children.
 66 */
 67 Cal3D.CalBone.prototype.calculateState = function() {
 68 	// check if the bone was not touched by any active animation
 69 	if(this.m_accumulatedWeight == 0) {
 70 		// set the bone to the initial skeleton state
 71 		this.m_translation.assign(this.m_coreBone.getTranslation());
 72 		this.m_rotation.assign(this.m_coreBone.getRotation());
 73 	}
 74 
 75 	// get parent bone id
 76 	var parentId = this.m_coreBone.getParentId();
 77 
 78 	if(parentId == -1) {
 79 		// no parent, this means absolute state == relative state
 80 		this.m_translationAbsolute.assign(this.m_translation);
 81 		this.m_rotationAbsolute.assign(this.m_rotation);
 82 	}
 83 	else {
 84 		// get the parent bone
 85 		var parent = this.m_skeleton.getBone(parentId);
 86 
 87 		// transform relative state with the absolute state of the parent
 88 		this.m_translationAbsolute.assign(this.m_translation);
 89 		this.m_translationAbsolute.multQuaternionLocal(parent.getRotationAbsolute());
 90 		this.m_translationAbsolute.addLocal(parent.getTranslationAbsolute());
 91 
 92 		this.m_rotationAbsolute.assign(this.m_rotation);
 93 		this.m_rotationAbsolute.multQuaternionLocal(parent.getRotationAbsolute());
 94 	}
 95 
 96 	// calculate the bone space transformation
 97 	this.m_translationBoneSpace.assign(this.m_coreBone.getTranslationBoneSpace());
 98 	this.m_translationBoneSpace.multQuaternionLocal(this.m_rotationAbsolute);
 99 	this.m_translationBoneSpace.addLocal(this.m_translationAbsolute);
100 
101 	this.m_rotationBoneSpace.assign(this.m_coreBone.getRotationBoneSpace());
102 	this.m_rotationBoneSpace.multQuaternionLocal(this.m_rotationAbsolute);
103 
104 	// Generate the vertex transform.  
105 	//TODO: If I ever add support for bone-scaling to Cal3D, this step will become significantly more complex.
106 	this.m_transformMatrix.assign(this.m_rotationBoneSpace);
107 
108 	// calculate all child bones
109 	var listChildId = this.m_coreBone.getListChildId();
110 	for(var iChildId=0; iChildId<listChildId.length; iChildId++) {
111 		this.m_skeleton.getBone(listChildId[iChildId]).calculateState();
112 	}
113 };

115 /**
116 	Clear the current state.<br />
117 	This function clears the current state (absolute translation and rotation) of the bone instance and all its children.
118 */
119 Cal3D.CalBone.prototype.clearState = function() {
120 	this.m_accumulatedWeight = 0;
121 	this.m_accumulatedWeightAbsolute = 0;
122 };

133 /**
134 	Reset the bone to its core state.<br />
135 	This function changes the state of the bone to its default non-animated position and orientation. Child bones are unaffected and 
136 	may be animated independently.
137 */
138 Cal3D.CalBone.prototype.setCoreState = function() {
139 	// set the bone to the initial skeleton state
140 	this.m_translation.assign(this.m_coreBone.getTranslation());
141 	this.m_rotation.assign(this.m_coreBone.getRotation());
142 
143 	// set the appropriate weights
144 	this.m_accumulatedWeightAbsolute = 1;
145 	this.m_accumulatedWeight = 1;
146 
147 	this.calculateState();
148 };
149 
150 /**
151 	Reset the bone and children to core states.<br />
152 	This function changes the state of the bone to its default non-animated position and orientation. All child bones are also set in this manner.
153 */
154 Cal3D.CalBone.prototype.setCoreStateRecursive = function() {
155 	// set the bone to the initial skeleton state
156 	this.m_translation.assign(this.m_coreBone.getTranslation());
157 	this.m_rotation.assign(this.m_coreBone.getRotation());
158 
159 	// set the appropriate weights
160 	this.m_accumulatedWeightAbsolute = 1;
161 	this.m_accumulatedWeight = 1;
162 
163 	// set core state for all child bones
164 	var listChildId = this.m_coreBone.getListChildId();
165 	for(var iChildId=0; iChildId<listChildId.length; iChildId++) {
166 		this.m_skeleton.getBone(listChildId[iChildId]).setCoreStateRecursive();
167 	}
168 
169 	this.calculateState() ;
170 };

257 /**
258 	Lock the current state.<br />
259 	This function locks the current state (absolute translation and rotation) of the bone instance and all its children.
260 */
261 Cal3D.CalBone.prototype.lockState = function() {
262 	// clamp accumulated weight
263 	if(this.m_accumulatedWeightAbsolute > 1 - this.m_accumulatedWeight) {
264 		this.m_accumulatedWeightAbsolute = 1 - this.m_accumulatedWeight;
265 	}
266 
267 	if(this.m_accumulatedWeightAbsolute > 0) {
268 		if(this.m_accumulatedWeight == 0) {
269 			// it is the first state, so just copy it into the bone state
270 			this.m_translation.assign(this.m_translationAbsolute);
271 			this.m_rotation.assign(this.m_rotationAbsolute);
272 			this.m_accumulatedWeight = this.m_accumulatedWeightAbsolute;
273 		}
274 		else {
275 			// it is not the first state, so blend all attributes
276 			var factor = this.m_accumulatedWeightAbsolute / (this.m_accumulatedWeight + this.m_accumulatedWeightAbsolute);
277 
278 			this.m_translation.blend(factor, this.m_translationAbsolute);
279 			this.m_rotation.blend(factor, this.m_rotationAbsolute);
280 			this.m_accumulatedWeight += this.m_accumulatedWeightAbsolute;
281 		}
282 
283 		this.m_accumulatedWeightAbsolute = 0;
284 	}
285 };

296 /**
297 	Calculate the bounding box of the bone.<br />
298 	This function calculates the bounding box of the bone instance.
299 */
300 Cal3D.CalBone.prototype.calculateBoundingBox = function() {
301 	if(!this.getCoreBone().isBoundingBoxPrecomputed())
302 		return;
303 
304 	var dir = new Cal3D.CalVector(1, 0, 0);
305 	dir.multMatrixLocal(this.getTransformMatrix());
306 	this.m_boundingBox.plane[0].setNormal(dir);
307 
308 	dir.assign(-1, 0, 0);
309 	dir.multMatrixLocal(this.getTransformMatrix());
310 	this.m_boundingBox.plane[1].setNormal(dir);
311 
312 	dir.assign(0, 1, 0);
313 	dir.multMatrixLocal(this.getTransformMatrix());
314 	this.m_boundingBox.plane[2].setNormal(dir);
315 
316 	dir.assign(0, -1, 0);
317 	dir.multMatrixLocal(this.getTransformMatrix());
318 	this.m_boundingBox.plane[3].setNormal(dir);
319 
320 	dir.assign(0, 0, 1);
321 	dir.multMatrixLocal(this.getTransformMatrix());
322 	this.m_boundingBox.plane[4].setNormal(dir);
323 
324 	dir.assign(0, 0, -1);
325 	dir.multMatrixLocal(this.getTransformMatrix());
326 	this.m_boundingBox.plane[5].setNormal(dir);
327 
328 	var position = new Cal3D.CalVector;
329 	for(var i=0; i<6; i++) {
330 		this.getCoreBone().getBoundingData(i, position);
331 
332 		position.multMatrixLocal(this.getTransformMatrix());
333 		position.addLocal(this.getTranslationBoneSpace());
334 
335 		for(var planeId=0; planeId<6; planeId++) {
336 			if(this.m_boundingBox.plane[planeId].eval(position) < 0) {
337 				this.m_boundingBox.plane[planeId].setPosition(position);
338 			}
339 		}
340 	}
341 };
