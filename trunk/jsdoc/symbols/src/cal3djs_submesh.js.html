 18 /**
 19 	@class CalSubmesh
 20 */
 21 Cal3D.CalSubmesh = function(coreSubmesh) {
 22 	this.m_coreSubmesh = coreSubmesh;
 23 	this.m_vectorMorphTargetWeight = new Array(coreSubmesh.getCoreSubMorphTargetCount());
 24 	this.m_vectorVertex = [];
 25 	this.m_vectorNormal = [];
 26 	this.m_vectorvectorTangentSpace = [];
 27 	this.m_vectorFace = new Array(coreSubmesh.getFaceCount());
 28 	this.m_vectorPhysicalProperty = [];
 29 	this.m_vertexCount = 0;
 30 	this.m_faceCount = 0;
 31 	this.m_coreMaterialId = -1;
 32 	this.m_bInternalData = false;
 33 
 34 	// initialize faces
 35 	for(var faceId=0; faceId<this.m_vectorFace.length; faceId++) {
 36 		this.m_vectorFace[faceId] = new Cal3D.CalSubmesh.Face;
 37 	}
 38 
 39 	// set the initial lod level
 40 	this.setLodLevel(1);
 41 
 42 	// set the initial morph target weights
 43 	var morphTargetCount = coreSubmesh.getCoreSubMorphTargetCount();
 44 	for(var morphTargetId=0; morphTargetId<morphTargetCount; morphTargetId++) {
 45 		this.m_vectorMorphTargetWeight[morphTargetId] = 0;
 46 	}
 47 
 48 	// check if the submesh instance must handle the vertex and normal data internally
 49 	if(coreSubmesh.getSpringCount() > 0) {
 50 		var vertexCount = coreSubmesh.getVertexCount();
 51 
 52 		this.m_vectorVertex = new Array(vertexCount);
 53 		this.m_vectorNormal = new Array(vertexCount);
 54 
 55 		this.m_vectorvectorTangentSpace = new Array(coreSubmesh.getVectorVectorTangentSpace().length);
 56 
 57 		this.m_vectorPhysicalProperty = new Array(vertexCount);
 58 
 59 		// get the vertex vector of the core submesh
 60 		var vectorVertex = coreSubmesh.getVectorVertex();
 61 
 62 		// copy the data from the core submesh as default values
 63 		for(var vertexId=0; vertexId<vertexCount; vertexId++) {
 64 			// copy the vertex data
 65 			this.m_vectorVertex[vertexId] = new Cal3D.CalVector(vectorVertex[vertexId].position);
 66 			var physicalProperty = new Cal3D.CalSubmesh.PhysicalProperty;
 67 			physicalProperty.position.assign(vectorVertex[vertexId].position);
 68 			physicalProperty.positionOld.assign(vectorVertex[vertexId].position);
 69 			this.m_vectorPhysicalProperty[vertexId] = physicalProperty;
 70 
 71 			// copy the normal data
 72 			this.m_vectorNormal[vertexId] = new Cal3D.CalVector(vectorVertex[vertexId].normal);
 73 		}
 74 
 75 		this.m_bInternalData = true;
 76 	}
 77 };

106 /**
107 	Get the face data.<br />
108 	This function returns the face data (vertex indices) of the submesh instance. The LOD setting of the submesh instance is taken into account.
109 	@param {Array} faceBuffer The user-provided buffer where the face data is written to. This is an output parameter.
110 	@returns {number} The number of faces written to the buffer.
111 */
112 Cal3D.CalSubmesh.prototype.getFaces = function(faceBuffer) {
113 	// copy the face vector to the face buffer
114 	for(var i=0, j=0; i<this.m_faceCount; i++, j+=3) {
115 		var faceVertId = this.m_vectorFace[i].vertexId;
116 		faceBuffer[j    ] = faceVertId[0];
117 		faceBuffer[j + 1] = faceVertId[1];
118 		faceBuffer[j + 2] = faceVertId[2];
119 	}
120 
121 	return this.m_faceCount;
122 };

201 /**
202 	Set the current LOD level.<br />
203 	This function sets the current LOD level of the submesh instance.
204 	@param {number} lodLevel The LOD level in the range [0.0, 1.0].
205 */
206 Cal3D.CalSubmesh.prototype.setLodLevel = function(lodLevel) {
207 	// clamp the lod level to [0.0, 1.0]
208 	if(lodLevel < 0) 
209 		lodLevel = 0;
210 	if(lodLevel > 1) 
211 		lodLevel = 1;
212 
213 	// get the lod count of the core submesh
214 	var lodCount = this.m_coreSubmesh.getLodCount();
215 
216 	// calculate the target lod count
217 	lodCount = ~~((1 - lodLevel) * lodCount);
218 
219 	// calculate the new number of vertices
220 	this.m_vertexCount = this.m_coreSubmesh.getVertexCount() - lodCount;
221 
222 	// get face vector of the core submesh
223 	var vectorFace = this.m_coreSubmesh.getVectorFace();
224 
225 	// get face vector of the core submesh
226 	var vectorVertex = this.m_coreSubmesh.getVectorVertex();
227 
228 	// calculate the new number of faces
229 	this.m_faceCount = vectorFace.length;
230 
231 	for(var vertexId=vectorVertex.length-1; vertexId>=this.m_vertexCount; vertexId--) {
232 		this.m_faceCount -= vectorVertex[vertexId].faceCollapseCount;
233 	}
234 
235 	// fill the face vector with the collapsed vertex ids
236 	for(var faceId=0; faceId<this.m_faceCount; faceId++) {
237 		for(var vertexId=0; vertexId<3; vertexId++) {
238 			// get the vertex id
239 			var collapsedVertexId = vectorFace[faceId].vertexId[vertexId];
240 
241 			// collapse the vertex id until it fits into the current lod level
242 			// FIXME: this may cause an infinit loop
243 			while(collapsedVertexId >= this.m_vertexCount) {
244 				collapsedVertexId = vectorVertex[collapsedVertexId].collapseId;
245 			}
246 
247 			// store the collapse vertex id in the submesh face vector
248 			this.m_vectorFace[faceId].vertexId[vertexId] = collapsedVertexId;
249 		}
250 	}
251 };

263 /**
264 	Enable (and calculate) or disable the storage of tangent spaces.<br />
265 	This function enables or disables the storage of tangent space bases.
266 	@param {number} mapId The ID of the texture map.
267 	@param {boolean} enabled Set to true to enable the tangent spaces, false to disable it.
268 	@returns {boolean} true if succeeded; false if any error happened.
269 */
270 Cal3D.CalSubmesh.prototype.enableTangents = function(mapId, enabled) {
271 	if(!this.m_coreSubmesh.enableTangents(mapId, enabled))
272 		return false;
273 
274 	if(!this.m_bInternalData)
275 		return true;
276 
277 	if(!enabled) {
278 		this.m_vectorvectorTangentSpace[mapId] = [];
279 		return true;
280 	}
281 
282 	this.m_vectorvectorTangentSpace[mapId] = new Array(this.m_coreSubmesh.getVertexCount());
283 
284 	// get the tangent space vector of the core submesh
285 	var vectorTangentSpace = this.m_coreSubmesh.getVectorVectorTangentSpace()[mapId];
286 
287 	// copy the data from the core submesh as default values
288 	var vertexCount = this.m_coreSubmesh.getVertexCount();
289 	for(var vertexId=0; vertexId<vertexCount; vertexId++) {
290 		// copy the tangent space data
291 		var tangentSpace = new Cal3D.CalSubmesh.TangentSpace();
292 		tangentSpace.tangent.assign(vectorTangentSpace[vertexId].tangent);
293 		tangentSpace.crossFactor = vectorTangentSpace[vertexId].crossFactor;
294 		this.m_vectorvectorTangentSpace[mapId][vertexId] = tangentSpace;
295 	}
296 
297 	return true;
298 };

322 /**
323 	Get weight of the base vertices.<br />
324 	@returns {number} The weight of the base vertices.
325 */
326 Cal3D.CalSubmesh.prototype.getBaseWeight = function() {
327 	var baseWeight = 1;
328 	var morphTargetCount = this.getMorphTargetWeightCount();
329 	for(var morphTargetId=0; morphTargetId<morphTargetCount; morphTargetId++) {
330 		baseWeight -= this.m_vectorMorphTargetWeight[morphTargetId];
331 	}
332 
333 	return baseWeight;
334 };
