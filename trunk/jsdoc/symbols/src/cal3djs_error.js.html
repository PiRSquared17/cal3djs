 31 	FILE_NOT_FOUND:				4, 
 32 	INVALID_FILE_FORMAT:		5,

 83 /**
 84 	Dump all information about the last error to the standard output.<br />
 85 */
 86 Cal3D.CalError.printLastError = function() {
 87 	throw 'not implemented error';
 88 };
 89 
 90 /**
 91 	Set all the information about the last error.<br />
 92 	@param {number} code The code of the last error.
 93 	@param {string} filename The name of the source file where the last error occured.
 94 	@param {number} line The line number in the source file where the last error occured.
 95 	@param {string} text The supplementary description of the last error.
 96 */
 97 Cal3D.CalError.setLastError = function(code, filename, line, text) {
 98 	if(code >= Cal3D.CalError.Code.MAX_ERROR_CODE)
 99 		code = Cal3D.CalError.Code.INTERNAL;
100 
101 	Cal3D.CalError.m_lastErrorCode = code;
102 	Cal3D.CalError.m_lastErrorFile = (filename != undefined) ? filename : '';
103 	Cal3D.CalError.m_lastErrorLine = (line != undefined) ? line : -1;
104 	Cal3D.CalError.m_lastErrorText = (text != undefined) ? text: '';
105 };
106 
107 /**
108 	Get description of a given error.<br />
109 	@param {number} code The error code.
110 	@returns {string} The description of the given error code.
111 */
112 Cal3D.CalError.getErrorDescription = function(code) {
113 	switch(code)
114 	{
115 	case Cal3D.CalError.Code.OK:                         return 'No error found';
116 	case Cal3D.CalError.Code.INTERNAL:                   return 'Internal error';
117 	case Cal3D.CalError.Code.INVALID_HANDLE:             return 'Invalid handle as argument';
118 	case Cal3D.CalError.Code.MEMORY_ALLOCATION_FAILED:   return 'Memory allocation failed';
119 	case Cal3D.CalError.Code.FILE_NOT_FOUND:             return 'File not found';
120 	case Cal3D.CalError.Code.INVALID_FILE_FORMAT:        return 'Invalid file format';
121 	case Cal3D.CalError.Code.FILE_PARSER_FAILED:         return 'Parser failed to process file';
122 	case Cal3D.CalError.Code.INDEX_BUILD_FAILED:         return 'Building of the index failed';
123 	case Cal3D.CalError.Code.NO_PARSER_DOCUMENT:         return 'There is no document to parse';
124 	case Cal3D.CalError.Code.INVALID_ANIMATION_DURATION: return 'The duration of the animation is invalid';
125 	case Cal3D.CalError.Code.BONE_NOT_FOUND:             return 'Bone not found';
126 	case Cal3D.CalError.Code.INVALID_ATTRIBUTE_VALUE:    return 'Invalid attribute value';
127 	case Cal3D.CalError.Code.INVALID_KEYFRAME_COUNT:     return 'Invalid number of keyframes';
128 	case Cal3D.CalError.Code.INVALID_ANIMATION_TYPE:     return 'Invalid animation type';
129 	case Cal3D.CalError.Code.FILE_CREATION_FAILED:       return 'Failed to create file';
130 	case Cal3D.CalError.Code.FILE_WRITING_FAILED:        return 'Failed to write to file';
131 	case Cal3D.CalError.Code.INCOMPATIBLE_FILE_VERSION:  return 'Incompatible file version';
132 	case Cal3D.CalError.Code.NO_MESH_IN_MODEL:           return 'No mesh attached to the model';
133 	case Cal3D.CalError.Code.BAD_DATA_SOURCE:            return 'Cannot read from data source';
134 	case Cal3D.CalError.Code.NULL_BUFFER:                return 'Memory buffer is null';
135 	case Cal3D.CalError.Code.INVALID_MIXER_TYPE:         return 'The CalModel mixer is not a CalMixer instance';
136 	default:											 return 'Unknown error';
137 	}
138 };
139 
140 /**
141 	Get description of the last error.<br />
142 	@returns {string} The description of the last error.
143 */
144 Cal3D.CalError.getLastErrorDescription = function() {
145 	return Cal3D.CalError.getErrorDescription(Cal3D.CalError.getLastErrorCode());
146 };
