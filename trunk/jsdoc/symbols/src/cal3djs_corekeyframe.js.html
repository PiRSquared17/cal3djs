 18 /**
 19 	@class CalCoreKeyframe
 20 */
 21 Cal3D.CalCoreKeyframe = function() {
 22 	this.m_time = 0;
 23 	this.m_translation = new Cal3D.CalVector;
 24 	this.m_rotation = new Cal3D.CalQuaternion;
 25 };
 26 
 27 /**
 28 	Create the core keyframe instance<br />
 29 	@returns {boolean} true if succeeded; false if failed.
 30 */
 31 Cal3D.CalCoreKeyframe.prototype.create = function() {
 32 	return true;
 33 };
 34 
 35 /**
 36 	Destroy the core keyframe instance.
 37 */
 38 Cal3D.CalCoreKeyframe.prototype.destroy = function() {
 39 };
 40 
 41 /**
 42 	Get the rotation.<br />
 43 	This function returns the rotation of the core keyframe instance.
 44 	@returns {Cal3D.CalQuaternion} The rotation as quaternion.
 45 */
 46 Cal3D.CalCoreKeyframe.prototype.getRotation = function() {
 47 	return this.m_rotation;
 48 };
 49 
 50 /**
 51 	Get the translation.<br />
 52 	This function returns the translation of the core keyframe instance.
 53 	@returns {Cal3D.CalVector} The translation as vector.
 54 */
 55 Cal3D.CalCoreKeyframe.prototype.getTranslation = function() {
 56 	return this.m_translation;
 57 };
 58 
 59 /**
 60 	Get the time.<br />
 61 	This function returns the time of the core keyframe instance.
 62 	@returns {number} The time in seconds.
 63 */
 64 Cal3D.CalCoreKeyframe.prototype.getTime = function() {
 65 	return this.m_time;
 66 };
 67 
 68 /**
 69 	Set the rotation.<br />
 70 	This function sets the rotation of the core keyframe instance.
 71 	@param {Cal3D.CalQuaternion} rotation The rotation to be set.
 72 */
 73 Cal3D.CalCoreKeyframe.prototype.setRotation = function(rotation) {
 74 	this.m_rotation.assign(rotation);
 75 };
 76 
 77 /**
 78 	Set the translation.<br />
 79 	This function sets the translation of the core keyframe instance.
 80 	@param {Cal3D.CalVector} translation The translation to be set.
 81 */
 82 Cal3D.CalCoreKeyframe.prototype.setTranslation = function(translation) {
 83 	this.m_translation.assign(translation);
 84 };
 85 
 86 /**
 87 	Set the time.<br />
 88 	This function sets the time of the core keyframe instance.
 89 	@param {number} time The time to be set in seconds.
 90 */
 91 Cal3D.CalCoreKeyframe.prototype.setTime = function(time) {
 92 	this.m_time = time;
 93 };
