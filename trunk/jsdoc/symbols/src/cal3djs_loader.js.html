 83 /**
 84 	Load an entire Cal3D model from a given config file.<br />
 85 	@param {string} url The url of the config file.
 86 	@param {object} callback An object which implements the loader callback interface to retrieve the modle loaded and/or the error information and/or the loading progress information.
 87 */
 88 Cal3D.CalLoader.loadModelFromConfigFile = function(url, callback) {
 89 	var xhr = new XMLHttpRequest;
 90 	xhr.open('GET', url, true);
 91 	xhr.overrideMimeType('text/plain; charset=x-user-defined');
 92 
 93 	xhr.onreadystatechange = function() {
 94 		if(this.readyState == 4) {
 95 			if(this.status == 200 || this.status == 0) {
 96 				var cfg = Cal3D.CalLoader.parseConfigFile(this.responseText);
 97 
 98 				if(cfg.skeletons.length == 0) {
 99 					Cal3D.CalError.setLastError(Cal3D.CalError.Code.FILE_NOT_FOUND, 'loader.js');
100 					if(callback && callback.onerror) {
101 						callback.onerror(Cal3D.CalError.Code.FILE_NOT_FOUND, 'skeleton file');
102 					}
103 					return;
104 				}
105 
106 				var resourceCount = 1 /* the one and only skeleton file */ + cfg.animations.length + cfg.meshes.length + cfg.materials.length;
107 				var resources = {
108 					skeletons: {}, 
109 					animations: {}, 
110 					meshes: {}, 
111 					materials: {}
112 				};
113 
114 				var setupModel = function() {
115 					// create a new core model instance
116 					var coreModel = new Cal3D.CalCoreModel('dummy');
117 
118 					// set core skeleton
119 					coreModel.setCoreSkeleton(resources.skeletons[cfg.path + cfg.skeletons[0]]);
120 
121 					// set core animations
122 					for(var i=0; i<cfg.animations.length; i++) {
123 						coreModel.addCoreAnimation(resources.animations[cfg.path + cfg.animations[i]]);
124 					}
125 
126 					// set core meshes
127 					for(var i=0; i<cfg.meshes.length; i++) {
128 						coreModel.addCoreMesh(resources.meshes[cfg.path + cfg.meshes[i]]);
129 					}
130 
131 					// set core materials
132 					for(var i=0; i<cfg.materials.length; i++) {
133 						coreModel.addCoreMaterial(resources.materials[cfg.path + cfg.materials[i]]);
134 					}
135 
136 					// make one material thread for each material
137 					// FIXME: maybe this is not the right way to do it, while this is the only thing 
138 					//        we can do here as we haven't got any further information about the model.
139 					for(var materialId=0; materialId<cfg.materials.length; materialId++) {
140 						// create the a material thread
141 						coreModel.createCoreMaterialThread(materialId);
142 
143 						// initialize the material thread
144 						coreModel.setCoreMaterialId(materialId, 0, materialId);
145 					}
146 
147 					// calculate bounding boxes
148 					coreModel.getCoreSkeleton().calculateBoundingBoxes(coreModel);
149 
150 					// create a new model instance
151 					var model = new Cal3D.CalModel(coreModel);
152 
153 					// attach all meshes to the model
154 					for(var meshId=0; meshId<cfg.meshes.length; meshId++) {
155 						model.attachMesh(meshId);
156 					}
157 
158 					// set the material set of the whole model
159 					model.setMaterialSet(0);
160 
161 					if(callback && callback.onload) {
162 						callback.onload({ model: model, path: cfg.path, scale: cfg.scale }, url);
163 					}
164 				};
165 
166 				var animationCB = {};
167 				animationCB.onload = function(coreAnimation, url) {
168 					resources.animations[url] = coreAnimation;
169 					if(--resourceCount == 0)
170 						setupModel();
171 				};
172 				animationCB.onerror = function(code, url) {
173 					if(callback && callback.onerror) {
174 						callback.onerror(code, url);
175 					}
176 				};
177 
178 				var meshCB = {};
179 				meshCB.onload = function(coreMesh, url) {
180 					resources.meshes[url] = coreMesh;
181 					if(--resourceCount == 0)
182 						setupModel();
183 				};
184 				meshCB.onerror = function(code, url) {
185 					if(callback && callback.onerror) {
186 						callback.onerror(code, url);
187 					}
188 				};
189 
190 				var materialCB = {};
191 				materialCB.onload = function(coreMaterial, url) {
192 					resources.materials[url] = coreMaterial;
193 					if(--resourceCount == 0)
194 						setupModel();
195 				};
196 				materialCB.onerror = function(code, url) {
197 					if(callback && callback.onerror) {
198 						callback.onerror(code, url);
199 					}
200 				};
201 
202 				var skeletonCB = {};
203 				skeletonCB.onload = function(coreSkeleton, url) {
204 					resources.skeletons[url] = coreSkeleton;
205 
206 					for(var i=0; i<cfg.animations.length; i++) {
207 						Cal3D.CalLoader.loadCoreAnimationFromFile(cfg.path + cfg.animations[i], coreSkeleton, animationCB);
208 					}
209 
210 					for(var i=0; i<cfg.meshes.length; i++) {
211 						Cal3D.CalLoader.loadCoreMeshFromFile(cfg.path + cfg.meshes[i], meshCB);
212 					}
213 
214 					for(var i=0; i<cfg.materials.length; i++) {
215 						Cal3D.CalLoader.loadCoreMaterialFromFile(cfg.path + cfg.materials[i], materialCB);
216 					}
217 
218 					if(--resourceCount == 0)
219 						setupModel();
220 				};
221 				skeletonCB.onerror = function(code, url) {
222 					if(callback && callback.onerror) {
223 						callback.onerror(code, url);
224 					}
225 				};
226 
227 				Cal3D.CalLoader.loadCoreSkeletonFromFile(cfg.path + cfg.skeletons[0], skeletonCB);
228 			}
229 		}
230 	};
231 
232 	xhr.onerror = function() {
233 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.FILE_NOT_FOUND, 'loader.js', 0, url);
234 		if(callback && callback.onerror) {
235 			callback.onerror(Cal3D.CalError.Code.FILE_NOT_FOUND, url);
236 		}
237 	};
238 
239 	xhr.onprogress = function(evt) {
240 		if(callback && callback.onprogress) {
241 			callback.onprogress(evt.position / evt.totalSize, url);
242 		}
243 	};
244 
245 	xhr.send();
246 };

268 				var coreAnim = Cal3D.CalLoader.loadCoreAnimationFromData(this.responseText, skel);

298 /**
299 	Load a core animation instance from a given data buffer.<br />
300 	@param {string} data The data buffer from which to load the core animation instance.
301 	@param {Cal3D.CalCoreSkeleton} skel The core skeleton instance the core animation is attached to. This parameter is optional.
302 	@returns {Cal3D.CalCoreAnimation} The core animation instance loaded; null if any error happened.
303 */
304 Cal3D.CalLoader.loadCoreAnimationFromData = function(data, skel) {
305 	// check if this is a valid animation file
306 	if(data.substring(0, 3) != Cal3D.CalLibraryConstants.ANIMATION_FILE_MAGIC) {
307 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_FILE_FORMAT, 'loader.js');
308 		return null;
309 	}
310 
311 	var datasource = new Cal3D.CalBufferSource(data.substring(4, data.length));
312 	
313 	// check if the version is compatible with the library
314 	var version = datasource.readInteger();
315 	if( isNaN(version) || version < Cal3D.CalLibraryConstants.EARLIEST_COMPATIBLE_FILE_VERSION || 
316 		version > Cal3D.CalLibraryConstants.CURRENT_FILE_VERSION ) {
317 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_FILE_FORMAT, 'loader.js');
318 		return null;
319 	}
320 
321 	// create a new core animation instance
322 	var coreAnimation = new Cal3D.CalCoreAnimation;
323 
324 	// get the duration of the core animation
325 	var duration = datasource.readFloat();
326 	if(isNaN(duration)) {
327 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_FILE_FORMAT, 'loader.js');
328 		return null;
329 	}
330 
331 	// check for a valid duration
332 	if(duration <= 0) {
333 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_ANIMATION_DURATION, 'loader.js');
334 		return null;
335 	}
336 
337 	// set the duration in the core animation instance
338 	coreAnimation.setDuration(duration);
339 
340 	// read the number of tracks
341 	var trackCount = datasource.readInteger();
342 	if(isNaN(trackCount) || trackCount <= 0) {
343 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_FILE_FORMAT, 'loader.js');
344 		return null;
345 	}
346 
347 	// load all core tracks
348 	for(var trackId=0; trackId<trackCount; trackId++) {
349 		// load the core track
350 		var coreTrack = Cal3D.CalLoader.loadCoreTrack(datasource, skel, duration);
351 		if(!coreTrack) {
352 			Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_FILE_FORMAT, 'loader.js');
353 			return null;
354 		}
355 
356 		// add the core track to the core animation instance
357 		coreAnimation.addCoreTrack(coreTrack);
358 	}
359 
360 	return coreAnimation;
361 };

412 /**
413 	Load a core material instance from a given data buffer.<br />
414 	@param {string} data The data buffer from which to load the core material instance.
415 	@returns {Cal3D.CalCoreMaterial} The core material instance loaded; null if any error happened.
416 */
417 Cal3D.CalLoader.loadCoreMaterialFromData = function(data) {
418 	// check if this is a valid material file
419 	if(data.substring(0, 3) != Cal3D.CalLibraryConstants.MATERIAL_FILE_MAGIC) {
420 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_FILE_FORMAT, 'loader.js');
421 		return null
422 	}
423 
424 	var datasource = new Cal3D.CalBufferSource(data.substring(4, data.length));
425 
426 	// check if the version is compatible with the library
427 	var version = datasource.readInteger();
428 	if( isNaN(version) || version < Cal3D.CalLibraryConstants.EARLIEST_COMPATIBLE_FILE_VERSION || 
429 		version > Cal3D.CalLibraryConstants.CURRENT_FILE_VERSION ) {
430 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INCOMPATIBLE_FILE_VERSION, 'loader.js');
431 		return null;
432 	}
433 
434 	// allocate a new core material instance
435 	var coreMaterial = new Cal3D.CalCoreMaterial;
436 
437 	var colorBuffer = [0, 0, 0, 0];
438 
439 	// get the ambient color of the core material
440 	var ambientColor = new Cal3D.CalCoreMaterial.Color;
441 	datasource.readBytes(colorBuffer, colorBuffer.length);
442 	ambientColor.red   = colorBuffer[0];
443 	ambientColor.green = colorBuffer[1];
444 	ambientColor.blue  = colorBuffer[2];
445 	ambientColor.alpha = colorBuffer[3];
446 
447 	// get the diffuse color of the core material
448 	var diffuseColor = new Cal3D.CalCoreMaterial.Color;
449 	datasource.readBytes(colorBuffer, colorBuffer.length);
450 	diffuseColor.red   = colorBuffer[0];
451 	diffuseColor.green = colorBuffer[1];
452 	diffuseColor.blue  = colorBuffer[2];
453 	diffuseColor.alpha = colorBuffer[3];
454 
455 	// get the specular color of the core material
456 	var specularColor = new Cal3D.CalCoreMaterial.Color;
457 	datasource.readBytes(colorBuffer, colorBuffer.length);
458 	specularColor.red   = colorBuffer[0];
459 	specularColor.green = colorBuffer[1];
460 	specularColor.blue  = colorBuffer[2];
461 	specularColor.alpha = colorBuffer[3];
462 
463 	// get the shininess factor of the core material
464 	var shininess = datasource.readFloat();
465 
466 	// check if an error happened
467 	if(!datasource.ok()) {
468 		datasource.setError();
469 		return null;
470 	}
471 
472 	// set the colors and the shininess
473 	coreMaterial.setAmbientColor(ambientColor);
474 	coreMaterial.setDiffuseColor(diffuseColor);
475 	coreMaterial.setSpecularColor(specularColor);
476 	coreMaterial.setShininess(shininess);
477 
478 	// read the number of maps
479 	var mapCount = datasource.readInteger();
480 	if(isNaN(mapCount) || mapCount < 0) {
481 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_FILE_FORMAT, 'loader.js');
482 		return null;
483 	}
484 
485 	// reserve memory for all the material data
486 	if(!coreMaterial.reserve(mapCount)) {
487 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.MEMORY_ALLOCATION_FAILED, 'loader.js');
488 		return null;
489 	}
490 
491 	// load all maps
492 	for(var mapId=0; mapId<mapCount; mapId++) {
493 		var map = new Cal3D.CalCoreMaterial.Map;
494 
495 		// read the filename of the map
496 		map.filename = datasource.readString();
497 
498 		// initialize the user data to null
499 		map.userData = null;
500 
501 		// check if an error happened
502 		if(!datasource.ok()) {
503 			Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_FILE_FORMAT, 'loader.js');
504 			return null;
505 		}
506 
507 		// set map in the core material instance
508 		coreMaterial.setMap(mapId, map);
509 	}
510 
511 	return coreMaterial;
512 };

736 	rz = datasource.readFloat();
737 	rw = datasource.readFloat();

743 	tzBoneSpace = datasource.readFloat();

766 		}

786 	coreBone.setRotationBoneSpace(rotbs);

799 			Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_FILE_FORMAT, 'loader.js');

858 		datasource.setError();
859 		return null;
860 	}
861 
862 	// get the material thread id of the submesh
863 	var coreMaterialThreadId = datasource.readInteger();
864 
865 	// get the number of vertices
866 	var vertexCount = datasource.readInteger();
867 
868 	// get the number of faces
869 	var faceCount = datasource.readInteger();
870 
871 	// get the number of level-of-details
872 	var lodCount = datasource.readInteger();
873 
874 	// get the number of springs
875 	var springCount = datasource.readInteger();
876 
877 	// get the number of texture coordinates per vertex
878 	var textureCoordinateCount = datasource.readInteger();
879 
880 	// check if an error happened
881 	if(!datasource.ok()) {
882 		datasource.setError();
883 		return null;
884 	}
885 
886 	// allocate a new core submesh instance
887 	var coreSubmesh = new Cal3D.CalCoreSubmesh;
888 
889 	// set the LOD step count
890 	coreSubmesh.setLodCount(lodCount);
891 
892 	// set the core material id
893 	coreSubmesh.setCoreMaterialThreadId(coreMaterialThreadId);
894 
895 	// reserve memory for all the submesh data
896 	if(!coreSubmesh.reserve(vertexCount, textureCoordinateCount, faceCount, springCount)) {
897 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.MEMORY_ALLOCATION_FAILED, 'loader.js');
898 		return null;
899 	}
900 
901 	// load the tangent space enable flags.
902 	for(var textureCoordinateId=0; textureCoordinateId<textureCoordinateCount; textureCoordinateId++) {
903 		coreSubmesh.enableTangents(textureCoordinateId, false);
904 	}
905 
906 	// load all vertices and their influences
907 	for(var vertexId=0; vertexId<vertexCount; vertexId++) {
908 		var vertex = new Cal3D.CalCoreSubmesh.Vertex;
909 
910 		// load data of the vertex
911 		vertex.position.x = datasource.readFloat();
912 		vertex.position.y = datasource.readFloat();
913 		vertex.position.z = datasource.readFloat();
914 		vertex.normal.x = datasource.readFloat();
915 		vertex.normal.y = datasource.readFloat();
916 		vertex.normal.z = datasource.readFloat();
917 		vertex.collapseId = datasource.readInteger();
918 		vertex.faceCollapseCount = datasource.readInteger();
919 
920 		// check if an error happened
921 		if(!datasource.ok()) {
922 			datasource.setError();
923 			return null;
924 		}
925 
926 		// load all texture coordinates of the vertex
927 		for(var vartextureCoordinateId=0; textureCoordinateId<textureCoordinateCount; textureCoordinateId++) {
928 			var textureCoordinate = new Cal3D.CalCoreSubmesh.TextureCoordinate;
929 
930 			// load the texture coordinate
931 			textureCoordinate.u = datasource.readFloat();
932 			textureCoordinate.v = datasource.readFloat();
933 
934 			if(Cal3D.CalLoader.loadingMode & Cal3D.CalLoader.LOADER_INVERT_V_COORD) {
935 				textureCoordinate.v = 1 - textureCoordinate.v;
936 			}
937 
938 			// check if an error happened
939 			if(!datasource.ok()) {
940 				datasource.setError();
941 				return null;
942 			}
943 
944 			// set texture coordinate in the core submesh instance
945 			coreSubmesh.setTextureCoordinate(vertexId, textureCoordinateId, textureCoordinate);
946 		}
947 
948 		// get the number of influences
949 		var influenceCount = datasource.readInteger();
950 		if(isNaN(influenceCount) || influenceCount < 0) {
951 			datasource.setError();
952 			return null;
953 		}
954 
955 		// load all influences of the vertex
956 		for(var influenceId=0; influenceId<influenceCount; influenceId++) {
957 			// load data of the influence
958 			var influence = new Cal3D.CalCoreSubmesh.Influence;
959 			influence.boneId = datasource.readInteger();
960 			influence.weight = datasource.readFloat();
961 			vertex.vectorInfluence.push(influence);
962 
963 			// check if an error happened
964 			if(!datasource.ok()) {
965 				datasource.setError();
966 				return null;
967 			}
968 		}
969 
970 		// set vertex in the core submesh instance
971 		coreSubmesh.setVertex(vertexId, vertex);
972 
973 		// load the physical property of the vertex if there are springs in the core submesh
974 		if(springCount > 0) {
975 			var physicalProperty = new Cal3D.CalCoreSubmesh.PhysicalProperty;
976 
977 			// load data of the physical property
978 			physicalProperty.weight = datasource.readFloat();
979 
980 			// check if an error happened
981 			if(!datasource.ok()) {
982 				datasource.setError();
983 				return null;
984 			}
985 
986 			// set the physical property in the core submesh instance
987 			coreSubmesh.setPhysicalProperty(vertexId, physicalProperty);
988 		}
989 	}
990 
991 	// load all springs
992 	for(var springId=0; springId<springCount; springId++) {
993 		var spring = new Cal3D.CalCoreSubmesh.Spring;
994 
995 		// load data of the spring
996 		spring.vertexId[0] = datasource.readInteger();
997 		spring.vertexId[1] = datasource.readInteger();
998 		spring.springCoefficient = datasource.readFloat();
999 		spring.idleLength = datasource.readFloat();
1000 
1001 		// check if an error happened
1002 		if(!datasource.ok()) {
1003 			datasource.setError();
1004 			return null;
1005 		}
1006 
1007 		// set spring in the core submesh instance
1008 		coreSubmesh.setSpring(springId, spring);
1009 	}
1010 
1011 
1012 	// load all faces
1013 	var justOnce = 0;
1014 	var flipModel = false;
1015 	for(var faceId=0; faceId<faceCount; faceId++) {
1016 		var face = new Cal3D.CalCoreSubmesh.Face;
1017 
1018 		// load data of the face
1019 		face.vertexId[0] = datasource.readInteger();
1020 		face.vertexId[1] = datasource.readInteger();
1021 		face.vertexId[2] = datasource.readInteger();
1022 
1023 		// check if an error happened
1024 		if(!datasource.ok()) {
1025 			datasource.setError();
1026 			return null;
1027 		}
1028 
1029 		// check if left-handed coord system is used by the object
1030 		// can be done only once since the object has one system for all faces
1031 		if(justOnce == 0) {
1032 			// get vertexes of first face
1033 			var vectorVertex = coreSubmesh.getVectorVertex();
1034 			var v1 = vectorVertex[face.vertexId[0]];
1035 			var v2 = vectorVertex[face.vertexId[1]];
1036 			var v3 = vectorVertex[face.vertexId[2]];
1037 
1038 			var point1 = new Cal3D.CalVector(v1.position);
1039 			var point2 = new Cal3D.CalVector(v2.position);
1040 			var point3 = new Cal3D.CalVector(v3.position);
1041 
1042 			// gets vectors (v1-v2) and (v3-v2)
1043 			var vect1 = Cal3D.vectorSub(point1, point2);
1044 			var vect2 = Cal3D.vectorSub(point3, point2);
1045 
1046 			// calculates normal of face
1047 			var cross = Cal3D.vectorCross(vect1, vect2);
1048 			var faceNormal = Cal3D.vectorScalarDiv(cross, cross.length());
1049 
1050 			// compare the calculated normal with the normal of a vertex
1051 			var maxNorm = new Cal3D.CalVector(v1.normal);
1052 
1053 			// if the two vectors point to the same direction then the poly needs flipping
1054 			// that is if the dot product > 0 it needs flipping
1055 			if (Cal3D.vectorDot(faceNormal, maxNorm) > 0)
1056 				flipModel = true;
1057 
1058 			// flip the winding order if the loading flags request it
1059 			if (Cal3D.CalLoader.loadingMode & Cal3D.CalLoader.LOADER_FLIP_WINDING)
1060 				flipModel = !flipModel;
1061 
1062 			justOnce = 1;
1063 		}
1064 
1065 
1066 		// flip if needed
1067 		if(flipModel) {
1068 			var tmp = face.vertexId[1];
1069 			face.vertexId[1] = face.vertexId[2];
1070 			face.vertexId[2] = tmp;
1071 		}
1072 
1073 		// set face in the core submesh instance
1074 		coreSubmesh.setFace(faceId, face);
1075 	}
1076 
1077 	return coreSubmesh;
1078 };

1799 				return null;

1803 			vertex.position.y = parseFloat(r[2]);
1804 			vertex.position.z = parseFloat(r[3]);

1815 				return null;

1833

1835 			}
1836 			else {
1837 				vertex.collapseId = -1;
1838 				vertex.faceCollapseCount = 0;
1839 			}
1840

1843 
1844 			// load all texture coordinates of the vertex

1872 			}
1873

1875 			var influenceCount = parseInt(vertexElem.getAttribute('NUMINFLUENCES'));
1876 			if(influenceCount < 0) {
1877 				Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_FILE_FORMAT, 'loader.js');
1878 				return null;
1879 			}
1880 
1881 			var influenceElem = texcoordElem;
1882 
1883 			// load all influences of the vertex
1884 			for(var influenceId=0; influenceId<influenceCount; influenceId++) {
1885 				if(!influenceElem || influenceElem.tagName != 'INFLUENCE') {
1886 					Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_FILE_FORMAT, 'loader.js');
1887 					return null;
1888 				}

1894 
1895 				// go to next influence
1896 				influenceElem = influenceElem.nextElementSibling;
1897 			}

1903 
1904 			// load the physical property of the vertex if there are springs in the core submesh
1905 			if(springCount > 0) {
1906 				if(!physiqueElem || physiqueElem.tagName != 'PHYSIQUE') {
1907 					Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_FILE_FORMAT, 'loader.js');
1908 					return null;
1909 				}
1910

1912 				physicalProperty.weight = parseFloat(physiqueElem.textContent);
1913

1915 				coreSubmesh.setPhysicalProperty(vertexId, physicalProperty);          
1916 			}
1917 
1918 			// go to next vertex
1919 			vertexElem = vertexElem.nextElementSibling;
1920 		}
1921 
1922 		var springElem = vertexElem;
1923 
1924 		// load all springs
1925 		for(var springId=0; springId<springCount; springId++) {
1926 			if(!springElem || springElem.tagName != 'SPRING') {
1927 				Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_FILE_FORMAT, 'loader.js');
1928 				return null;
1929 			}

1932 
1933 			var r = Cal3D.CalLoader.pattern_int2.exec(springElem.getAttribute('VERTEXID'));

1935 				Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_FILE_FORMAT, 'loader.js');
1936 				return null;
1937 			}
1938

1940 			spring.vertexId[1] = parseInt(r[2]);
1941 			spring.springCoefficient = parseFloat(springElem.getAttribute('COEF'));
1942 			spring.idleLength = parseFloat(springElem.getAttribute('LENGTH'));
1943 
1944 			// set spring in the core submesh instance
1945 			coreSubmesh.setSpring(springId, spring);
1946 
1947 			// go to next spring
1948 			springElem = springElem.nextElementSibling;
1949 		}
1950 
1951 		var faceElem = springElem;
1952 
1953 		// load all faces
1954 		for(var faceId=0; faceId<faceCount; faceId++) {
1955 			if(!faceElem || faceElem.tagName != 'FACE') {
1956 				Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_FILE_FORMAT, 'loader.js');
1957 				return null;
1958 			}
1959 
1960 			var face = new Cal3D.CalCoreSubmesh.Face;
1961 
1962 			// load data of the face
1963 			var r = Cal3D.CalLoader.pattern_int3.exec(faceElem.getAttribute('VERTEXID'));
1964 			if(!r) {
1965 				Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_FILE_FORMAT, 'loader.js');
1966 				return null;
1967 			}
1968 
1969 			face.vertexId[0] = parseInt(r[1]);
1970 			face.vertexId[1] = parseInt(r[2]);
1971 			face.vertexId[2] = parseInt(r[3]);
1972 
1973 			coreSubmesh.setFace(faceId, face);
1974 
1975 			// go to next face
1976 			faceElem = faceElem.nextElementSibling;
1977 		}
1978 
1979 		// add the core submesh to the core mesh instance
1980 		coreMesh.addCoreSubmesh(coreSubmesh);
1981 
1982 		// go to next submesh
1983 		submeshElem = submeshElem.nextElementSibling;
1984 	}

1986 	return coreMesh;
