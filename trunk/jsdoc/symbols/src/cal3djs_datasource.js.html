 18 /**
 19 	@class CalBufferSource
 20 */
 21 Cal3D.CalBufferSource = function(inputBuffer) {
 22 	this.m_buffer = inputBuffer;
 23 	this.m_offset = 0;
 24 };
 25 
 26 /**
 27 	Check if the buffer source is in good state.<br />
 28 	@returns {boolean} true if the buffer source is in good state; false if not.
 29 */
 30 Cal3D.CalBufferSource.prototype.ok = function() {
 31 	return this.m_buffer != null;
 32 };
 33 
 34 /**
 35 	Set the error code and message related to a the buffer source.
 36 */
 37 Cal3D.CalBufferSource.prototype.setError = function() {
 38 	Cal3D.CalError.setLastError(Cal3D.CalError.Code.NULL_BUFFER, 'datasource.js');
 39 };
 40 
 41 /**
 42 	Read a given number of bytes from the buffer source.<br />
 43 	@param {Array} The array where the bytes are stored into.
 44 	@param {number} count The number of bytes to be read.
 45 	@returns {number} The number of bytes that was actually read.
 46 */
 47 Cal3D.CalBufferSource.prototype.readBytes = function(buffer, count) {
 48 	if(!this.ok() || !buffer || count <= 0)
 49 		return 0;
 50 
 51 	if(this.m_offset + count > this.m_buffer.length)
 52 		count = this.m_buffer.length - this.m_offset;
 53 
 54 	for(var i=0; i<count; i++) {
 55 		buffer[i] = this.m_buffer[this.m_offset + i].charCodeAt(0) & 0xff;
 56 	}
 57 	this.m_offset += count;
 58 
 59 	return count;
 60 };
 61 
 62 /**
 63 	Read a 32-bits little endian float number from the buffer source.<br />
 64 	@returns {number} The float value. NaN will be returned if any error happened.
 65 */
 66 Cal3D.CalBufferSource.prototype.readFloat = function() {
 67 	if(!this.ok())
 68 		return NaN;
 69 
 70 	if(this.m_offset + 4 > this.m_buffer.length)
 71 		return NaN;
 72 
 73 	var mLen = 23;
 74 	var eLen = 8;		// 4 * 8 - 23 - 1
 75 	var eMax = 255;		// (1 << eLen) - 1;
 76 	var eBias = 127;	// eMax >> 1;
 77 
 78 	var i = 3; 
 79 	var d = -1; 
 80 	var s = this.m_buffer[this.m_offset + i].charCodeAt(0) & 0xff; 
 81 	i += d; 
 82 	var bits = -7;
 83 	var e = s & ((1 << (-bits)) - 1);
 84 	s >>= -bits;
 85 	bits += eLen
 86 	while(bits > 0) {
 87 		e = e * 256 + (this.m_buffer[this.m_offset + i].charCodeAt(0) & 0xff);
 88 		i += d;
 89 		bits -= 8;
 90 	}
 91 
 92 	var m = e & ((1 << (-bits)) - 1);
 93 	e >>= -bits;
 94 	bits += mLen;
 95 	while(bits > 0) {
 96 		 m = m * 256 + (this.m_buffer[this.m_offset + i].charCodeAt(0) & 0xff);
 97 		 i += d;
 98 		 bits -= 8;
 99 	}
100 
101 	this.m_offset += 4;
102 
103 	switch(e) {
104 		case 0:		// 0 or denormalized number
105 			e = 1 - eBias;
106 			break;
107 		case eMax:	// NaN or +/-Infinity
108 			return m ? NaN : ((s ? -1 : 1) * Infinity);
109 		default:	// normalized number
110 			m = m + Math.pow(2, mLen);
111 			e = e - eBias;
112 			break;
113 	}
114 
115 	return (s ? -1 : 1) * m * Math.pow(2, e - mLen);
116 };
117 
118 /**
119 	Read a 32-bits little endian signed integer number from the buffer source.<br />
120 	@returns {number} The integer value. NaN will be returned if any error happened.
121 */
122 Cal3D.CalBufferSource.prototype.readInteger = function() {
123 	if(!this.ok())
124 		return NaN;
125 
126 	if(this.m_offset + 4 > this.m_buffer.length)
127 		return NaN;
128 
129 	var rv = 0, f = 1;
130 	for(var i=0; i<4; i++) {
131 		rv += ((this.m_buffer[this.m_offset + i].charCodeAt(0) & 0xff) * f);
132 		f *= 256;
133 	}
134 
135 	if(rv & 0x80000000)
136 		rv -= 0x100000000;
137 
138 	this.m_offset += 4;
139 
140 	return rv;
141 };
142 
143 /**
144 	Read a string from the buffer source.<br />
145 	@returns {string} The string value. An empty string will be returned if any error happened.
146 */
147 Cal3D.CalBufferSource.prototype.readString = function() {
148 	if(!this.ok())
149 		return '';
150 
151 	// read and skip the 'string length' 32-bits integer field
152 	var length = this.readInteger();
153 	if(isNaN(length) || length <= 0)
154 		return '';
155 
156 	// read out all characters other than the null-terminator('\0')
157 	var s = this.m_buffer.substring(this.m_offset, this.m_offset + length - 1);
158 	this.m_offset += length;
159 
160 	return s;
161 };
