 18 /**
 19 	@class CalCoreMorphAnimation
 20 */
 21 Cal3D.CalCoreMorphAnimation = function() {
 22 	this.m_vectorCoreMeshID = [];
 23 	this.m_vectorMorphTargetID = [];
 24 };
 25 
 26 /**
 27 	Add a core mesh ID and a morph target ID of that core mesh.<br />
 28 	@param {number} coreMeshID The core mesh ID to be added.
 29 	@param {number} morphTargetID The morph target ID to be added.
 30 	@returns {boolean} true if succeeded; false if any error happened.
 31 */
 32 Cal3D.CalCoreMorphAnimation.prototype.addMorphTarget = function(coreMeshID, morphTargetID) {
 33 	this.m_vectorCoreMeshID.push(coreMeshID);
 34 	this.m_vectorMorphTargetID.push(morphTargetID);
 35 	return true;
 36 };
 37 
 38 /**
 39 	Get the core mesh ID list.<br />
 40 	This function returns the list that contains all core mesh IDs of the core morph animation instance.
 41 	@returns {Array} The the core mesh ID list.
 42 */
 43 Cal3D.CalCoreMorphAnimation.prototype.getVectorCoreMeshID = function() {
 44 	return this.m_vectorCoreMeshID;
 45 };
 46 
 47 /**
 48 	Get the morph target ID list.<br />
 49 	This function returns the list that contains all morph target  IDs of the core morph animation instance.
 50 	@returns {Array} The morph target ID list.
 51 */
 52 Cal3D.CalCoreMorphAnimation.prototype.getVectorMorphTargetID = function() {
 53 	return this.m_vectorMorphTargetID;
 54 };
