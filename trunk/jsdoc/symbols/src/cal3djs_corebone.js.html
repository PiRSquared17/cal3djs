 21 Cal3D.CalCoreBone = function(name) {
 22 	this.m_name = name;
 23 	this.m_coreSkeleton = null;
 24 	this.m_parentId = -1;
 25 	this.m_listChildId = [];
 26 	this.m_translation = new Cal3D.CalVector;
 27 	this.m_rotation = new Cal3D.CalQuaternion;
 28 	this.m_translationAbsolute = new Cal3D.CalVector;
 29 	this.m_rotationAbsolute = new Cal3D.CalQuaternion;
 30 	this.m_translationBoneSpace = new Cal3D.CalVector;
 31 	this.m_rotationBoneSpace = new Cal3D.CalQuaternion;
 32 	this.m_userData = null;
 33 
 34 	this.m_boundingBox = new Cal3D.CalBoundingBox;
 35 	this.m_boundingPosition = [ 
 36 		new Cal3D.CalVector, 
 37 		new Cal3D.CalVector, 
 38 		new Cal3D.CalVector, 
 39 		new Cal3D.CalVector, 
 40 		new Cal3D.CalVector, 
 41 		new Cal3D.CalVector 
 42 	];
 43 	this.m_boundingBoxPrecomputed = false;
 44 };

 46 /**
 47 	Add a child ID.<br />
 48 	This function adds a core bone ID to the child ID list of the core bone instance.
 49 	@param {number} childId The ID of the core bone ID that shoud be added to the child ID list.s
 50 	@returns {boolean} false if any error happeded.

 52 Cal3D.CalCoreBone.prototype.addChildId = function(childId) {
 53 	this.m_listChildId.push(childId);
 54 	return true;

 57 /**
 58 	Calculate the current state.<br />
 59 	This function calculates the current state (absolute translation and rotation) of the core bone instance and all its children.
 60 */
 61 Cal3D.CalCoreBone.prototype.calculateState = function() {
 62 	if(this.m_parentId == -1) {
 63 		// no parent, this means absolute state == relative state
 64 		this.m_translationAbsolute.assign(this.m_translation);
 65 		this.m_rotationAbsolute.assign(this.m_rotation);
 66 	}
 67 	else {
 68 		// get the parent bone
 69 		var parent = this.m_coreSkeleton.getCoreBone(this.m_parentId);
 70 
 71 		// transform relative state with the absolute state of the parent
 72 		this.m_translationAbsolute.assign(this.m_translation);
 73 		this.m_translationAbsolute.multQuaternionLocal(parent.getRotationAbsolute());
 74 		this.m_translationAbsolute.addLocal(parent.getTranslationAbsolute());
 75 
 76 		this.m_rotationAbsolute.assign(this.m_rotation);
 77 		this.m_rotationAbsolute.multQuaternionLocal(parent.getRotationAbsolute());
 78 	}
 79 
 80 	// calculate all child bones
 81 	for(var iChildId=0; iChildId<this.m_listChildId.length; iChildId++) {
 82 		this.m_coreSkeleton.getCoreBone(this.m_listChildId[iChildId]).calculateState();
 83 	}
 84 };

248 /**
249 	Calculate the bounding box.<br />
250 	This function Calculates the bounding box of the core bone instance.
251 	@param {Cal3D.CalCoreModel} coreModel The core model needed to retrieve vertices.
252 */
253 Cal3D.CalCoreBone.prototype.calculateBoundingBox = function(coreModel) {
254 	var boneId = this.m_coreSkeleton.getCoreBoneId(this.m_name);
255 	var bBoundsComputed = false;
256 
257 	var rot = new Cal3D.CalQuaternion(this.m_rotationBoneSpace);
258 	rot.invert();
259 
260 	var dir = new Cal3D.CalVector(1, 0, 0);
261 	dir.multQuaternionLocal(rot);
262 	this.m_boundingBox.plane[0].setNormal(dir);
263 
264 	dir.assign(-1, 0, 0);
265 	dir.multQuaternionLocal(rot);
266 	this.m_boundingBox.plane[1].setNormal(dir);
267 
268 	dir.assign(0, 1, 0);
269 	dir.multQuaternionLocal(rot);
270 	this.m_boundingBox.plane[2].setNormal(dir);
271 
272 	dir.assign(0, -1, 0);
273 	dir.multQuaternionLocal(rot);
274 	this.m_boundingBox.plane[3].setNormal(dir);
275 
276 	dir.assign(0, 0, 1);
277 	dir.multQuaternionLocal(rot);
278 	this.m_boundingBox.plane[4].setNormal(dir);
279 
280 	dir.assign(0, 0, -1);
281 	dir.multQuaternionLocal(rot);
282 	this.m_boundingBox.plane[5].setNormal(dir);
283 
284 	var coreMeshCount = coreModel.getCoreMeshCount();
285 	for(var meshId=0; meshId<coreMeshCount; meshId++) {
286 		var coreMesh = coreModel.getCoreMesh(meshId);
287 
288 		var coreSubmeshCount = coreMesh.getCoreSubmeshCount();
289 		for(var submeshId=0; submeshId<coreSubmeshCount; submeshId++) {
290 			var coreSubmesh = coreMesh.getCoreSubmesh(submeshId);
291 
292 			if(coreSubmesh.getSpringCount() == 0) {
293 				var vectorVertex = coreSubmesh.getVectorVertex();
294 				for(var vertexId=0; vertexId<vectorVertex.length; vertexId++) {
295 					var vertexInfluenceCount = vectorVertex[vertexId].vectorInfluence.length;
296 					for(var influenceId=0; influenceId<vertexInfluenceCount; influenceId++) {
297 						if(vectorVertex[vertexId].vectorInfluence[influenceId].boneId == boneId && vectorVertex[vertexId].vectorInfluence[influenceId].weight > 0.5) {
298 							for(var planeId=0; planeId<6; planeId++) {
299 								if(this.m_boundingBox.plane[planeId].eval(vectorVertex[vertexId].position) < 0) {
300 									this.m_boundingBox.plane[planeId].setPosition(vectorVertex[vertexId].position);
301 									this.m_boundingPosition[planeId].assign(vectorVertex[vertexId].position);
302 									bBoundsComputed = true;
303 								}
304 							}
305 						}
306 					}
307 				}	
308 			}
309 		}
310 	}
311 
312 	// to handle bones with no vertices assigned 
313 	if(!bBoundsComputed) {
314 		for(var planeId=0; planeId<6; planeId++) {
315 			this.m_boundingBox.plane[planeId].setPosition(this.m_translation); 
316 			this.m_boundingPosition[planeId].assign(this.m_translation); 
317 		} 
318 	} 
319 
320 	this.m_boundingBoxPrecomputed = true;
321 };

350 /**
351 	Scale the core bone.<br />
352 	This function rescale all the data that are in the core bone instance and in all its children.
353 	@param {number} factor The scale factor.
354 */
355 Cal3D.CalCoreBone.prototype.scale = function(factor) {
356 	this.m_translation.multScalarLocal(factor);
357 	this.m_translationAbsolute.multScalarLocal(factor);
358 	this.m_translationBoneSpace.multScalarLocal(factor);
359 
360 	// calculate all child bones
361 	for(var iChildId=0; iChildId<this.m_listChildId.length; iChildId++)	{
362 		this.m_coreSkeleton.getCoreBone(this.m_listChildId[iChildId]).scale(factor);
363 	}
364 };
