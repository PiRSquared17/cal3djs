 18 /**
 19 	@class CalModel
 20 */
 21 Cal3D.CalModel = function(coreModel) {
 22 	this.m_coreModel = coreModel;
 23 	this.m_skeleton = new Cal3D.CalSkeleton(coreModel.getCoreSkeleton());
 24 	this.m_mixer = new Cal3D.CalMixer(this);
 25 	this.m_morphTargetMixer = new Cal3D.CalMorphTargetMixer(this);
 26 	this.m_physique = new Cal3D.CalPhysique(this);
 27 	this.m_springSystem = new Cal3D.CalSpringSystem(this);
 28 	this.m_renderer = new Cal3D.CalRenderer(this);
 29 	this.m_userData = null;
 30 	this.m_vectorMesh = [];
 31 	this.m_boundingBox = new Cal3D.CalBoundingBox;
 32 };
 33 
 34 /**
 35 	Attach a mesh to the model.<br />
 36 	This function attachs a mesh to the model instance.
 37 	@param {number} coreMeshId The ID of the mesh to be attached.
 38 	@returns {boolean} true if succeeded; false if the mesh does not exist.
 39 */
 40 Cal3D.CalModel.prototype.attachMesh = function(coreMeshId) {
 41 	// check if the id is valid
 42 	if(coreMeshId < 0 || coreMeshId >= this.m_coreModel.getCoreMeshCount()) {
 43 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_HANDLE, 'model.js');
 44 		return false;
 45 	}
 46 
 47 	// get the core mesh
 48 	var coreMesh = this.m_coreModel.getCoreMesh(coreMeshId);
 49 
 50 	// check if the mesh is already attached
 51 	for(var meshId=0; meshId<this.m_vectorMesh.length; meshId++) {
 52 		// check if we found the matching mesh
 53 		if(this.m_vectorMesh[meshId].getCoreMesh() == coreMesh) {
 54 			// mesh is already active so nothing to do
 55 			return true;
 56 		}
 57 	}
 58 
 59 	// create a new mesh instance
 60 	var mesh = new Cal3D.CalMesh(coreMesh);
 61 
 62 	// set model in the mesh instance
 63 	mesh.setModel(this);
 64 
 65 	// insert the new mesh into the active list
 66 	this.m_vectorMesh.push(mesh);
 67 
 68 	return true;
 69 };

 71 /**
 72 	Detach a mesh from the model.<br />
 73 	This function detaches a mesh from the model instance.
 74 	@param {number} coreMeshId The ID of the mesh to be detached.
 75 	@returns {boolean} true if succeeded; false if the mesh does not exist or any error happened.
 76 */
 77 Cal3D.CalModel.prototype.detachMesh = function(coreMeshId) {
 78 	// check if the id is valid
 79 	if(coreMeshId < 0 || coreMeshId >= this.m_coreModel.getCoreMeshCount()) {
 80 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_HANDLE, 'model.js');
 81 		return false;
 82 	}
 83 
 84 	// get the core mesh
 85 	var coreMesh = this.m_coreModel.getCoreMesh(coreMeshId);
 86 
 87 	// find the mesh for the given id
 88 	for(var meshId=0; meshId<this.m_vectorMesh.length; meshId++) {
 89 		// get the mesh
 90 		var mesh = this.m_vectorMesh[meshId];
 91 
 92 		// check if we found the matching mesh
 93 		if(mesh.getCoreMesh() == coreMesh) {
 94 			// erase the mesh out of the active mesh list
 95 			this.m_vectorMesh.splice(meshId, 1);
 96 
 97 			return true;
 98 		}
 99 	}
100 
101 	return false;
102 };

113 /**
114 	Get an attached mesh.<br />
115 	This function returns the attached mesh with the given core mesh ID.
116 	@param {number} coreMeshId The core mesh ID of the mesh.
117 	@returns {Cal3D.CalMesh} The mesh; null if the mesh does not exist.
118 */
119 Cal3D.CalModel.prototype.getMesh = function(coreMeshId) {
120 	// check if the id is valid
121 	if(coreMeshId < 0 || coreMeshId >= this.m_coreModel.getCoreMeshCount()) {
122 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_HANDLE, 'model.js');
123 		return null;
124 	}
125 
126 	// get the core mesh
127 	var coreMesh = this.m_coreModel.getCoreMesh(coreMeshId);
128 
129 	// search the mesh
130 	for(var meshId=0; meshId<this.m_vectorMesh.length; meshId++) {
131 		// check if we found the matching mesh
132 		if(this.m_vectorMesh[meshId].getCoreMesh() == coreMesh) {
133 			return this.m_vectorMesh[meshId];
134 		}
135 	}
136 
137 	return null;
138 };

140 /**
141 	Get the mixer.<br />
142 	If a mixer that is not an instance of Cal3D.CalMixer was set with the CalModel.setAbstractMixer() method, an INVALID_MIXER_TYPE error 
143 	is set and null is returned.
144 	@returns {Cal3D.CalMixer} The mixer; null if the mixer is not set yet or not an instance of Cal3D.CalMixer.
145 */
146 Cal3D.CalModel.prototype.getMixer = function() {
147 	if(!this.m_mixer)
148 		return null;
149 
150 	if(!this.m_mixer.isDefaultMixer()) {
151 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_MIXER_TYPE, 'model.js');
152 		return null;
153 	}
154 
155 	return this.m_mixer;
156 };

219 /**
220 	Get the global bounding box of the model.<br />
221 	This function returns the global bounding box of the model.
222 	@param {boolean} precision indicate if the function need to calculate a correct bounding box.
223 	@returns {Cal3D.CalBoundingBox} The bounding box of the model.
224 */
225 Cal3D.CalModel.prototype.getBoundingBox = function(precision) {
226 	var norm = new Cal3D.CalVector(1, 0, 0);	
227 	this.m_boundingBox.plane[0].setNormal(norm);
228 
229 	norm.assign(-1, 0, 0);	
230 	this.m_boundingBox.plane[1].setNormal(norm);
231 
232 	norm.assign(0, 1, 0);	
233 	this.m_boundingBox.plane[2].setNormal(norm);
234 
235 	norm.assign(0, -1, 0);	
236 	this.m_boundingBox.plane[3].setNormal(norm);
237 
238 	norm.assign(0, 0, 1);	
239 	this.m_boundingBox.plane[4].setNormal(norm);
240 
241 	norm.assign(0, 0, -1);	
242 	this.m_boundingBox.plane[5].setNormal(norm);
243 
244 	if(precision)
245 		this.m_skeleton.calculateBoundingBoxes();
246 
247 	var vectorBone = this.m_skeleton.getVectorBone();
248 	var bboxPoints = [  new Cal3D.CalVector, new Cal3D.CalVector, new Cal3D.CalVector, new Cal3D.CalVector, 
249 						new Cal3D.CalVector, new Cal3D.CalVector, new Cal3D.CalVector, new Cal3D.CalVector ];
250 	for(var boneId=0; boneId<vectorBone.length; boneId++) {
251 		var bone = vectorBone[boneId];
252 
253 		// If it's just an approximation that are needed then
254 		// we just compute the bounding box from the skeleton
255 		if(!precision || !bone.getCoreBone().isBoundingBoxPrecomputed()) {
256 			var translation =  bone.getTranslationAbsolute();
257 
258 			for(var planeId=0; planeId<6; planeId++) {
259 				if(this.m_boundingBox.plane[planeId].eval(translation) < 0) {
260 					this.m_boundingBox.plane[planeId].setPosition(translation);
261 				}
262 			}
263 		}
264 		else {
265 			var localBoundingBox = bone.getBoundingBox();
266 			localBoundingBox.computePoints(bboxPoints);
267 
268 			for(var i=0; i<8; i++) {
269 				for(var planeId=0; planeId<6; planeId++) {
270 					if(this.m_boundingBox.plane[planeId].eval(bboxPoints[i]) < 0) {
271 						this.m_boundingBox.plane[planeId].setPosition(bboxPoints[i]);
272 					}
273 				}
274 			}				
275 		}
276 	}
277 
278 	return this.m_boundingBox;
279 };

299 /**
300 	Set the current LOD level.<br />
301 	This function sets the current LOD level of all attached meshes.
302 	@param {number} lodLevel The LOD level in the range [0.0, 1.0].
303 */
304 Cal3D.CalModel.prototype.setLodLevel = function(lodLevel) {
305 	// set the lod level in all meshes
306 	for(var meshId=0; meshId<this.m_vectorMesh.length; meshId++) {
307 		// set the lod level in the mesh
308 		this.m_vectorMesh[meshId].setLodLevel(lodLevel);
309 	}

334 /**
335 	Update the model.<br />
336 	This function updates the model instance for a given amount of time.
337 	@param {number} deltaTime The elapsed time in seconds since the last update.
338 */
339 Cal3D.CalModel.prototype.update = function(deltaTime) {
340 	this.m_mixer.updateAnimation(deltaTime);
341 	this.m_mixer.updateSkeleton();
342 
343 	this.m_morphTargetMixer.update(deltaTime);
344 	this.m_physique.update();
345 	this.m_springSystem.update(deltaTime);
346 };

348 /**
349 	Disable internal data (and thus the springs system).
350 */
351 Cal3D.CalModel.prototype.disableInternalData = function() {
352 	// disable internal data in all meshes
353 	for(var meshId=0; meshId<this.m_vectorMesh.length; meshId++) {
354 		// disable internal data in the mesh
355 		this.m_vectorMesh[meshId].disableInternalData();
356 	}
357 };
