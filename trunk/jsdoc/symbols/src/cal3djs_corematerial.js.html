 18 /**
 19 	@class CalCoreMaterial
 20 */
 21 Cal3D.CalCoreMaterial = function() {
 22 	this.m_ambientColor = new Cal3D.CalCoreMaterial.Color;
 23 	this.m_diffuseColor = new Cal3D.CalCoreMaterial.Color;
 24 	this.m_specularColor = new Cal3D.CalCoreMaterial.Color;
 25 	this.m_shininess = 0;
 26 	this.m_vectorMap = [];
 27 	this.m_userData = null;
 28 	this.m_name = '';
 29 	this.m_filename = '';
 30 };

 50 /**
 51 	Get the number of texture maps.<br />
 52 	This function returns the number of texture mapss in the core material instance.
 53 	@returns {number} The number of texture maps.
 54 */
 55 Cal3D.CalCoreMaterial.prototype.getMapCount = function() {
 56 	return this.m_vectorMap.length;
 57 };
 58 
 59 /**
 60 	Get a specified map texture filename.<br />
 61 	This function returns the texture filename for a specified map ID of the core material instance.
 62 	@param {number} mapId The ID of the texture map.
 63 	@returns {string} The filename of the map texture. An empty string will be returned if the map does not exist.
 64 */
 65 Cal3D.CalCoreMaterial.prototype.getMapFilename = function(mapId) {
 66 	if(mapId < 0 || mapId >= this.m_vectorMap.length) {
 67 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_HANDLE, 'corematerial.js');
 68 		return '';
 69 	}
 70 
 71 	var map = this.m_vectorMap[mapId];
 72 	if(!map)
 73 		return '';
 74 
 75 	return map.filename;
 76 };
 77 
 78 /**
 79 	Get user data of a specified textue map.<br />
 80 	This function returns the user data stored in the specified map of the core material instance.
 81 	@param {number} mapId The ID of the texture map.
 82 	@returns {object} The user data stored in the specified map; null if the map does not exist.
 83 */
 84 Cal3D.CalCoreMaterial.prototype.getMapUserData = function(mapId) {
 85 	if(mapId < 0 || mapId >= this.m_vectorMap.length) {
 86 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_HANDLE, 'corematerial.js');
 87 		return null;
 88 	}
 89 
 90 	var map = this.m_vectorMap[mapId];
 91 	if(!map)
 92 		return null;
 93 
 94 	return map.userData;
 95 };

133 /**
134 	Allocate space for the maps<br />
135 	This function reserves space for the maps of the core material instance.
136 	@param {number} mapCount The number of maps that this core material should holds.
137 	@returns {boolean} true, if succeeded; false if failed.
138 */

168 /**
169 	Set a specified texture map.<br />
170 	This function sets a specified map in the core material instance.
171 	@param {number} mapId The ID of the map.
172 	@param {Cal3D.CalCoreMaterial.Map} map The map to be set.
173 	@returns {boolean} true if succeeded; false if the map does not exist.
174 */
175 Cal3D.CalCoreMaterial.prototype.setMap = function(mapId, map) {
176 	if(mapId < 0 || mapId >= this.m_vectorMap.length)
177 		return false;
178 
179 	this.m_vectorMap[mapId] = map;
180 	return true;
181 };
182 
183 /**
184 	Store specified map user data.<br />
185 	This function stores user data in a specified map of the core material instance.
186 	@param {number} mapId The ID of the map.
187 	@param {object} userData The map to be set.
188 	@returns {boolean} true if succeeded; false if the map does not exist.
189 */
190 Cal3D.CalCoreMaterial.prototype.setMapUserData = function(mapId, userData) {
191 	if(mapId < 0 || mapId >= this.m_vectorMap.length)
192 		return false;
193 
194 	var map = this.m_vectorMap[mapId];
195 	if(!map)
196 		return false;
197 
198 	map.userData = userData;
199 	return true;
200 };
