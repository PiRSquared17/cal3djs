 18 /**
 19 	@class CalAbstractMixer
 20 
 21 	CalAbstractMixer defines the API that CalModel relies on for blending and scheduling animations.<br />
 22 	A third party mixer must implement this API in order to register itself with the CalModel.setAbstractMixer method. 
 23 	The default mixer (CalMixer) is an example of such implementation.<br /><br />
 24 
 25 	Cal3D expects a mixer to handle two tasks: scheduling and blending. Scheduling refers to everything related to time such 
 26 	as when an animation must run or when it must stop. Blending defines how concurrent animations influence each other: for 
 27 	instance walking and waving.<br /><br />
 28 
 29 	If CalMixer proves to be insufficient for the applications needs, an alternate mixer can be implemented and used without 
 30 	notifying Cal3D in any way. It is not mandatory to subclass CalAbstractMixer. However, when chosing this path, one must 
 31 	also avoid using the CalModel.update method because it would use the default mixer instantiated by the CalModel.create method 
 32 	with undesirable side effects. In addition libraries based on Cal3D (think NebulaDevice or OpenSceneGraph adapters) are 
 33 	not aware of these constraints and will keep calling the CalModel.update method of CalModel regardless.<br /><br />
 34 
 35 	Subclassing CalAbstractMixer when implementing an alternate mixer therefore provides a better integration with Cal3D and 
 36 	libraries that rely on CalModel. However, an additional effort is required in order to achieve compatibility with libraries or 
 37 	applications that rely on the CalMixer API (i.e. that use methods such as blendCycle or executeAction).  The CalMixer API is not 
 38 	meant to be generic and there is no reason to define an abstract class that specifies it. For historical reasons and because 
 39 	CalMixer is the default mixer, some applications and libraries (think Soya or CrystalSpace) depend on it. If they want to switch 
 40 	to a scheduler with extended capabilities it might be painfull for them to learn a completely different API. A scheduler with the 
 41 	ambition to obsolete CalMixer should therefore provide an API compatible with it to ease the migration process.<br /><br />
 42 
 43 	Short summary, if you want to write a new mixer:<br /><br />
 44 
 45 	1) An external mixer: ignore CalAbstractMixer and implement a mixer of your own. Avoid calling CalModel.update and any library or 
 46 	application that will call it behind your back. Avoid libraries and applications that rely on the default mixer CalMixer, as returned 
 47 	by CalModel.getMixer.<br /><br />
 48 
 49 	2) A mixer registered in Cal3D: subclass CalAbstractMixer, register it with CalModel.setAbstractMixer.  Avoid libraries and 
 50 	applications that rely on the default mixer CalMixer, as returned by CalModel.getMixer. CalModel.getMixer will return a null 
 51 	pointer if CalModel.setAbstractMixer was called to set a mixer that is not an instance of CalMixer.<br /><br />
 52 
 53 	3) A CalMixer replacement: same as 2) and provide a subclass of your own mixer that implements the CalMixer API so that existing 
 54 	applications can switch to it by calling CalModel.getAbstractMixer instead of CalModel.getMixer. The existing code using the CalMixer 
 55 	methods will keep working and the developper will be able to switch to a new API when convenient.
 56 */
 57 Cal3D.CalAbstractMixer = function() {
 58 };

 98 	@class CalMixer
 99 */
100 Cal3D.CalMixer = function(model) {
101 	this.m_model = model;
102 	this.m_vectorAnimation = new Array(model.getCoreModel().getCoreAnimationCount());
103 	this.m_listAnimationAction = [];
104 	this.m_listAnimationCycle = [];
105 	this.m_animationTime = 0;
106 	this.m_animationDuration = 0;
107 	this.m_timeFactor = 1;
108 };

121 /**
122 	Interpolate the weight of an animation cycle.<br />
123 	This function interpolates the weight of an animation cycle to a new value in a given amount of time. If the specified 
124 	animation cycle is not active yet, it is activated.
125 	@param {number} id The ID of the animation cycle to be blended.
126 	@param {number} weight The weight to interpolate the animation cycle to.
127 	@param {number} delay The time in seconds until the new weight should be reached.
128 	@returns {boolean} true if succeeded; false if any error happened.
129 */
130 Cal3D.CalMixer.prototype.blendCycle = function(id, weight, delay) {
131 	if(id < 0 || id >= this.m_vectorAnimation.length) {
132 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_HANDLE, 'mixer.js');
133 		return false;
134 	}
135 
136 	// get the animation for the given id
137 	var animation = this.m_vectorAnimation[id];
138 
139 	// create a new animation instance if it is not active yet
140 	if(!animation) {
141 		// take the fast way out if we are trying to clear an inactive animation
142 		if(weight == 0) 
143 			return true;
144 
145 		// get the core animation
146 		var coreAnimation = this.m_model.getCoreModel().getCoreAnimation(id);
147 		if(!coreAnimation) 
148 			return false;
149 
150 		// ensure that the animation's first and last key frame match for proper looping
151 		Cal3D.addExtraKeyframeForLoopedAnim(coreAnimation);
152 
153 		// create a new animation cycle instance
154 		var animationCycle = new Cal3D.CalAnimationCycle(coreAnimation);
155 
156 		// insert the new animation into the tables
157 		this.m_vectorAnimation[id] = animationCycle;
158 		this.m_listAnimationCycle.unshift(animationCycle);
159 
160 		// blend the animation
161 		return animationCycle.blend(weight, delay);
162 	}
163 
164 	// check if this is really an animation cycle instance
165 	if(animation.getType() != Cal3D.CalAnimation.Type.TYPE_CYCLE) {
166 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_ANIMATION_TYPE, 'mixer.js');
167 		return false;
168 	}
169 
170 	// clear the animation cycle from the active vector if the target weight is zero
171 	if(weight == 0) {
172 		this.m_vectorAnimation[id] = null;
173 	}
174 
175 	// cast it to an animation cycle
176 	var animationCycle = animation;
177 
178 	// blend the animation cycle
179 	animationCycle.blend(weight, delay);
180 	animationCycle.checkCallbacks(0, this.m_model);
181 
182 	return true;
183 };

185 /**
186 	Fade an animation cycle out.<br />
187 	This function fades an animation cycle out in a given amount of time.
188 	@param {number} id The ID of the animation cycle to be faded out.
189 	@param {number} delay The time in seconds until the the animation cycle is completely removed.
190 	@returns {boolean} true if succeeded; false if any error happened.
191 */
192 Cal3D.CalMixer.prototype.clearCycle = function(id, delay) {
193 	// check if the animation id is valid
194 	if(id < 0 || id >= this.m_vectorAnimation.length) {
195 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_HANDLE, 'mixer.js');
196 		return false;
197 	}
198 
199 	// get the animation for the given id
200 	var animation = this.m_vectorAnimation[id];
201 
202 	// we can only clear cycles that are active
203 	if(!animation) 
204 		return true;
205 
206 	// check if this is really a animation cycle instance
207 	if(animation.getType() != Cal3D.CalAnimation.Type.TYPE_CYCLE) {
208 		Cal3D.CalError.setLastError(Cal3D.CalError.Code.INVALID_ANIMATION_TYPE, 'mixer.js');
209 		return false;
210 	}
211 
212 	// clear the animation cycle from the active vector
213 	this.m_vectorAnimation[id] = null;
214 
215 	// cast it to an animation cycle
216 	var animationCycle = animation;
217 
218 	// set animation cycle to async state
219 	animationCycle.setAsync(this.m_animationTime, this.m_animationDuration);
220 
221 	// blend the animation cycle
222 	animationCycle.blend(0, delay);
223 	animationCycle.checkCallbacks(0, this.m_model);
224 
225 	return true;
226 };

228 /**
229 	Execute an animation action.<br />
230 	This function executes an animation action.
231 	@param {number} id The ID of the animation action to be execute.
232 	@param {number} delayIn  The time in seconds until the animation action reaches the full weight from the beginning of its execution.
233 	@param {number} delayOut The time in seconds in which the animation action reaches zero weight at the end of its execution.
234 	@param {number} weightTarget The weight to interpolate the animation action to.
235 	@param {boolean} autoLock This prevents the action from being reset and removed on the last keyframe if true.
236 	@returns {boolean} true if succeeded; false if any error happened.
237 */
238 Cal3D.CalMixer.prototype.executeAction = function(id, delayIn, delayOut, weightTarget, autoLock) {
239 	// get the core animation
240 	var coreAnimation = this.m_model.getCoreModel().getCoreAnimation(id);
241 	if(!coreAnimation)
242 		return false;
243 
244 	// create a new animation action instance
245 	var animationAction = new Cal3D.CalAnimationAction(coreAnimation);
246 
247 	// insert the new animation into the table
248 	this.m_listAnimationAction.unshift(animationAction);
249 
250 	// execute the animation
251 	animationAction.execute(delayIn, delayOut, weightTarget, autoLock);
252 	animationAction.checkCallbacks(0, this.m_model);
253 
254 	return true;
255 };

257 /**
258 	Clear an active animation action.<br />
259 	This function removes an animation action from the blend list. This is particularly useful with auto-locked actions on their last frame.
260 	@param {number} id The ID of the animation action to be removed.
261 	@returns {boolean} true if succeeded; false if the animation does not exist any error happened.
262 */
263 Cal3D.CalMixer.prototype.removeAction = function(id) {
264 	// get the core animation
265 	var coreAnimation = this.m_model.getCoreModel().getCoreAnimation(id);
266 	if(!coreAnimation)
267 		return false;
268 
269 	// update all active animation actions of this model
270 	var animationActionId = 0;
271 	while(animationActionId < this.m_listAnimationAction.length) {
272 		// find the specified action and remove it
273 		if(this.m_listAnimationAction[animationActionId].getCoreAnimation() == coreAnimation) {
274 			// found, so remove
275 			this.m_listAnimationAction[animationActionId].completeCallbacks(this.m_model);
276 			this.m_listAnimationAction.splice(animationActionId, 1);
277 			return true;
278 		}
279 
280 		animationActionId++;
281 	}
282 
283 	return false;
284 };

286 /**
287 	Update all active animations.<br />
288 	This function updates all active animations of the mixer instance for a given amount of time.
289 	@param {number} deltaTime The elapsed time in seconds since the last update.
290 */
291 Cal3D.CalMixer.prototype.updateAnimation = function(deltaTime) {
292 	// update the current animation time
293 	if(this.m_animationDuration == 0) {
294 		this.m_animationTime = 0;
295 	}
296 	else {
297 		this.m_animationTime += deltaTime * this.m_timeFactor;
298 		if(this.m_animationTime >= this.m_animationDuration || this.m_animationTime < 0) {
299 			this.m_animationTime = this.m_animationTime % this.m_animationDuration;
300 		}
301 		if (this.m_animationTime < 0) {
302 			this.m_animationTime += this.m_animationDuration;
303 		}
304 	}
305 
306 	// update all active animation actions of this model
307 	var animationActionId = 0;
308 	while(animationActionId < this.m_listAnimationAction.length) {
309 		var animationAction = this.m_listAnimationAction[animationActionId];
310 		// update and check if animation action is still active
311 		if(animationAction.update(deltaTime)) {
312 			animationAction.checkCallbacks(animationAction.getTime(), this.m_model);
313 			animationActionId++;
314 		}
315 		else {
316 			// animation action has ended, destroy and remove it from the animation list
317 			animationAction.completeCallbacks(this.m_model);
318 			this.m_listAnimationAction.splice(animationActionId, 1);
319 		}
320 	}
321 
322 	// TODO: update all active animation poses of this model
323 	// ...
324 
325 	// update the weight of all active animation cycles of this model
326 	var accumulatedWeight = 0;
327 	var accumulatedDuration = 0;
328 	var animationCycleId = 0;
329 	while(animationCycleId < this.m_listAnimationCycle.length) {
330 		var animationCycle = this.m_listAnimationCycle[animationCycleId];
331 		// update and check if animation cycle is still active
332 		if(animationCycle.update(deltaTime)) {
333 			// check if it is in sync. if yes, update accumulated weight and duration
334 			if(animationCycle.getState() == Cal3D.CalAnimation.State.STATE_SYNC) {
335 				accumulatedWeight += animationCycle.getWeight();
336 				accumulatedDuration += animationCycle.getWeight() * animationCycle.getCoreAnimation().getDuration();
337 			}
338 
339 			animationCycle.checkCallbacks(this.m_animationTime, this.m_model);
340 			animationCycleId++;
341 		}
342 		else {
343 			// animation cycle has ended, destroy and remove it from the animation list
344 			animationCycle.completeCallbacks(this.m_model);
345 			this.m_listAnimationCycle.splice(animationCycleId, 1);
346 		}
347 	}
348 
349 	// adjust the global animation cycle duration
350 	if(accumulatedWeight > 0) {
351 		this.m_animationDuration = accumulatedDuration / accumulatedWeight;
352 	}
353 	else {
354 		this.m_animationDuration = 0;
355 	}
356 };

358 /**
359 	Update the skeleton of the corresponding model to match the current animation state.
360 */
361 Cal3D.CalMixer.prototype.updateSkeleton = function() {
362 	// get the skeleton we need to update
363 	var skeleton = this.m_model.getSkeleton();
364 	if(!skeleton) 
365 		return;
366 
367 	// clear the skeleton state
368 	skeleton.clearState();
369 
370 	// get the bone vector of the skeleton
371 	var vectorBone = skeleton.getVectorBone();
372 
373 	var translation = new Cal3D.CalVector;
374 	var rotation = new Cal3D.CalQuaternion;
375 
376 	// loop through all animation actions
377 	for(var animationActionId=0; animationActionId<this.m_listAnimationAction.length; animationActionId++) {
378 		var animationAction = this.m_listAnimationAction[animationActionId];
379 		
380 		// get the core animation instance
381 		var coreAnimation = animationAction.getCoreAnimation();
382 
383 		// get the list of core tracks of above core animation
384 		var listCoreTrack = coreAnimation.getListCoreTrack();
385 
386 		// loop through all core tracks of the core animation
387 		for(var coreTrackId=0; coreTrackId<listCoreTrack.length; coreTrackId++) {
388 			// get the appropriate bone of the track
389 			var bone = vectorBone[ listCoreTrack[coreTrackId].getCoreBoneId() ];
390 
391 			// get the current translation and rotation
392 			listCoreTrack[coreTrackId].getState(animationAction.getTime(), translation, rotation);
393 
394 			// blend the bone state with the new state
395 			bone.blendState(animationAction.getWeight(), translation, rotation);
396 		}
397 	}
398 
399 	// lock the skeleton state
400 	skeleton.lockState();
401 
402 	// loop through all animation cycles
403 	for(var animationCycleId=0; animationCycleId<this.m_listAnimationCycle.length; animationCycleId++) {
404 		var animationCycle = this.m_listAnimationCycle[animationCycleId];
405 
406 		// get the core animation instance
407 		var coreAnimation = animationCycle.getCoreAnimation();
408 
409 		// calculate adjusted time
410 		var animationTime;
411 		if(animationCycle.getState() == Cal3D.CalAnimation.State.STATE_SYNC) {
412 			if(this.m_animationDuration == 0)
413 				animationTime = 0;
414 			else
415 				animationTime = this.m_animationTime * coreAnimation.getDuration() / this.m_animationDuration;
416 		}
417 		else {
418 			animationTime = animationCycle.getTime();
419 		}
420 
421 		// get the list of core tracks of above core animation
422 		var listCoreTrack = coreAnimation.getListCoreTrack();
423 
424 		// loop through all core tracks of the core animation
425 		for(var coreTrackId=0; coreTrackId<listCoreTrack.length; coreTrackId++) {
426 			// get the appropriate bone of the track
427 			var bone = vectorBone[ listCoreTrack[coreTrackId].getCoreBoneId() ];
428 
429 			// get the current translation and rotation
430 			listCoreTrack[coreTrackId].getState(animationTime, translation, rotation);
431 
432 			// blend the bone state with the new state
433 			bone.blendState(animationCycle.getWeight(), translation, rotation);
434 		}
435 	}
436 
437 	// lock the skeleton state
438 	skeleton.lockState();
439 
440 	// let the skeleton calculate its final state
441 	skeleton.calculateState();
442 };

525 /**
526 	Examine the given animation and if the first and last keyframe of a given track do not match up, the first key frame is duplicated 
527 	and added to the end of the track to ensure smooth looping.
528 	@private
529 */
530 Cal3D.addExtraKeyframeForLoopedAnim = function(coreAnimation) {
531 	var listCoreTrack = coreAnimation.getListCoreTrack();
532 	if(listCoreTrack.length == 0)
533 		return;
534 
535 	var coreTrackId = 0;
536 	if(!listCoreTrack[coreTrackId])
537 		return;
538 
539 	var lastKeyframe = listCoreTrack[coreTrackId].getCoreKeyframe( listCoreTrack[coreTrackId].getCoreKeyframeCount() - 1 );
540 	if(!lastKeyframe)
541 		return;
542 
543 	if(lastKeyframe.getTime() < coreAnimation.getDuration()) {
544 		for(coreTrackId=0; coreTrackId<listCoreTrack.length; coreTrackId++) {
545 			var coreTrack = listCoreTrack[coreTrackId];
546 
547 			var firstKeyframe = coreTrack.getCoreKeyframe(0);
548 			var newKeyframe = new Cal3D.CalCoreKeyframe;
549 
550 			newKeyframe.setTranslation(firstKeyframe.getTranslation());
551 			newKeyframe.setRotation(firstKeyframe.getRotation());
552 			newKeyframe.setTime(coreAnimation.getDuration());
553 
554 			coreTrack.addCoreKeyframe(newKeyframe);
555 		}
556 	}
557 };
