 18 /**
 19 	@class CalPlane
 20 	@private
 21 */
 22 Cal3D.CalPlane = function() {
 23 	this.a = 0;
 24 	this.b = 0;
 25 	this.c = 0;
 26 	this.d = 0;
 27 };
 28 
 29 Cal3D.CalPlane.prototype.eval = function(p) {
 30 	return p.x * this.a + p.y * this.b + p.z * this.c + this.d;
 31 };
 32 
 33 Cal3D.CalPlane.prototype.setPosition = function(p) {
 34 	this.d = -p.x * this.a - p.y * this.b - p.z * this.c;
 35 };
 36 
 37 Cal3D.CalPlane.prototype.setNormal = function(n) {
 38 	 this.a = n.x;
 39 	 this.b = n.y;
 40 	 this.c = n.z;
 41 	 this.d = -1e32;
 42 };
 43 
 44 Cal3D.CalPlane.prototype.dist = function(p) {
 45 	return Math.abs( (p.x * this.a + p.y * this.b + p.z * this.c + this.d) / Math.sqrt(this.a * this.a + this.b * this.b + this.c * this.c) );
 46 };

 50 /**
 51 	@class CalBoundingBox
 52 */
 53 Cal3D.CalBoundingBox = function() {
 54 	this.plane = [  
 55 		new Cal3D.CalPlane, 
 56 		new Cal3D.CalPlane, 
 57 		new Cal3D.CalPlane, 
 58 		new Cal3D.CalPlane, 
 59 		new Cal3D.CalPlane, 
 60 		new Cal3D.CalPlane
 61 	];
 62 };
 63 
 64 /**
 65 	Calculate the corner points of the bounding box.<br />
 66 	This function computes the 8 corner points of the bounding box.
 67 	@param {Array} points An array of 8 vectors to take the result. If null, the function will allocate one.
 68 	@returns {Array} An array that holds 8 vectors as the result.
 69 */
 70 Cal3D.CalBoundingBox.prototype.computePoints = function(points) {
 71 	if(!points || !(points instanceof Array) || points.length < 8)
 72 		points = new Array(8);
 73 
 74 	var m = new Cal3D.CalMatrix;
 75 
 76 	var plane = this.plane;	 
 77 	var p = 0;
 78 	for(var i=0; i<2; i++) {
 79 		for(var j=2; j<4; j++) {
 80 			for(var k=4; k<6; k++) {
 81 				var x,y,z;
 82 
 83 				m.dxdx = plane[i].a; m.dxdy = plane[i].b; m.dxdz = plane[i].c;        
 84 				m.dydx = plane[j].a; m.dydy = plane[j].b; m.dydz = plane[j].c;        
 85 				m.dzdx = plane[k].a; m.dzdy = plane[k].b; m.dzdz = plane[k].c;
 86 
 87 				var det = m.det();
 88 			   
 89 				if(det != 0) {
 90 					m.dxdx = -plane[i].d; m.dxdy = plane[i].b; m.dxdz = plane[i].c;        
 91 					m.dydx = -plane[j].d; m.dydy = plane[j].b; m.dydz = plane[j].c;        
 92 					m.dzdx = -plane[k].d; m.dzdy = plane[k].b; m.dzdz = plane[k].c;
 93 
 94 					x = m.det() / det;
 95 
 96 					m.dxdx = plane[i].a; m.dxdy = -plane[i].d; m.dxdz = plane[i].c;        
 97 					m.dydx = plane[j].a; m.dydy = -plane[j].d; m.dydz = plane[j].c;        
 98 					m.dzdx = plane[k].a; m.dzdy = -plane[k].d; m.dzdz = plane[k].c;
 99 
100 					y = m.det() / det;
101 
102 					m.dxdx = plane[i].a; m.dxdy = plane[i].b; m.dxdz = -plane[i].d;        
103 					m.dydx = plane[j].a; m.dydy = plane[j].b; m.dydz = -plane[j].d;        
104 					m.dzdx = plane[k].a; m.dzdy = plane[k].b; m.dzdz = -plane[k].d;
105 
106 					z = m.det() / det;
107 
108 					if(points[p])
109 						points[p].assign(x, y, z);
110 					else
111 						points[p] = new Cal3D.CalVector(x, y, z);
112 				}
113 				else {
114 					if(points[p])
115 						points[p].assign(0, 0, 0);
116 					else
117 						points[p] = new Cal3D.CalVector(0, 0, 0);
118 				}
119 
120 				p++;
121 			}
122 		}
123 	}
124 
125 	return points;
126 };
